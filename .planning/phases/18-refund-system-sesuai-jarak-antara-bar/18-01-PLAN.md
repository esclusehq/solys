```sql
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),

    reason TEXT,
    processed_at TIMESTAMPTZ,
    created_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    updated_at TIMESTAMPTZ NOT NULL DEFAULT NOW()
```

```rust
use chrono::{DateTime, Utc};

#[derive(Debug, Clone, Copy, PartialEq, Eq)]
pub enum RefundEligibility {
    Full,      // 0-7 days: full refund
    Prorated,  // 8-30 days: prorated amount
    None,      // 30+ days: no refund
}

#[derive(Debug, Clone, Copy, PartialEq, Eq)]
pub enum RefundStatus {
    Pending,
    Processed,
    Rejected,
}

pub fn calculate_eligibility(subscription_start: DateTime<Utc>, request_time: DateTime<Utc>) -> RefundEligibility {
    let days_elapsed = (request_time - subscription_start).num_days();
    match days_elapsed {
        0..=7 => RefundEligibility::Full,
        8..=30 => RefundEligibility::Prorated,
        _ => RefundEligibility::None,
    }
}

pub fn calculate_prorated_amount(monthly_price: f64, days_elapsed: i64) -> f64 {
    let daily_rate = monthly_price / 30.0;
    let remaining_days = (30 - days_elapsed).max(0);
    daily_rate * remaining_days as f64
}
```
