2. The implementation pattern:
```rust
// Add after the status update block (around line 117)
if server.status == "running" && status == "stopped" {
    // Fetch full server to get auto_restart flag
    if let Ok(Some(full_server)) = self.repository.find_by_id(&server.id).await {
        if full_server.auto_restart {
            tracing::warn!("Server {} detected as crashed, auto-restarting...", full_server.name);
            let executor = self.executor_factory.get_executor(&full_server);
            match executor.start(&full_server).await {
                Ok(_) => {
                    tracing::info!("Auto-restart triggered for {}", full_server.name);
                    // Update restart count
                    let mut updated = full_server;
                    updated.restart_count += 1;
                    let _ = self.repository.update(&updated).await;
                }
                Err(e) => {
                    tracing::error!("Auto-restart failed for {}: {}", full_server.name, e);
                }
            }
        }
    }
}
```
