    Create a new module at `agent/solys/src/crash_reporter.rs` that captures container crash forensic data using Bollard:

    ```rust
    use bollard::container::LogsOptions;
    use bollard::Docker;
    use futures_util::StreamExt;
    use uuid::Uuid;
    use anyhow::Result;

    /// Captures container exit code and last N lines of logs.
    /// Called when a managed container exits unexpectedly.
    pub async fn capture_crash_data(
        docker: &Docker,
        container_id: &str,
    ) -> Result<(i32, String)> {
        // 1. Inspect container to get exit code
        let inspect = docker.inspect_container(container_id, None).await?;
        let exit_code = inspect
            .state
            .and_then(|s| s.exit_code)
            .unwrap_or(-1);

        // 2. Read last 10 lines of logs (both stdout and stderr)
        let log_options = LogsOptions::<String> {
            stdout: true,
            stderr: true,
            tail: 10,
            ..Default::default()
        };

        let mut log_stream = docker.logs(container_id, Some(log_options));
        let mut log_lines: Vec<String> = Vec::new();

        while let Some(Ok(chunk)) = log_stream.next().await {
            log_lines.push(chunk.to_string());
        }

        let log_excerpt = log_lines.join("\n");

        // 3. Truncate to 4KB max to avoid flooding WebSocket (per Pitfall 2)
        let log_excerpt = if log_excerpt.len() > 4096 {
            format!("... (truncated) ...\n{}", &log_excerpt[log_excerpt.len() - 4000..])
        } else {
            log_excerpt
        };

        Ok((exit_code, log_excerpt))
    }
    ```

    The implementation details:

    - The event listener should have access to `ws_sender` (clone the sender) and `docker` client
    - Track which containers are managed by this agent (from registered servers or active containers)
    - Only report crashes for containers that were supposed to be running (not planned stops from backend commands)

    Pseudo-structure for the event listener:
    ```rust
    let ws_sender_clone = ws_sender.clone();
    let docker_clone = docker.clone();
    let managed_servers = managed_servers.clone(); // shared set of server_id → container_id

    tokio::spawn(async move {
        use bollard::system::EventsOptions;
        use bollard::system::EventStream;
        use futures_util::StreamExt;

        let mut events_stream = docker_clone.system_events::<&str>(None);
        
        while let Some(Ok(event)) = events_stream.next().await {
            if let Some(ref event_type) = event.typ {
                if event_type == "container" {
                    if let Some(ref action) = event.action {
                        if action == "die" {
                            if let Some(ref actor) = event.actor {
                                if let Some(ref attrs) = actor.attributes {
                                    if let Some(container_id) = actor.id.as_ref() {
                                        // Check if this container is managed
                                        if let Some(server_id) = find_server_id(container_id, &managed_servers) {
                                            if let Ok((exit_code, log_excerpt)) = crash_reporter::capture_crash_data(&docker_clone, container_id).await {
                                                let report = crash_reporter::build_crash_report(server_id, exit_code, log_excerpt);
                                                if let Ok(msg) = serde_json::to_string(&report) {
                                                    let _ = ws_sender_clone.send(Message::Text(msg.into())).await;
                                                }
                                            }
                                        }
                                    }
                                }
                            }
                        }
                    }
                }
            }
        }
    });
    ```
