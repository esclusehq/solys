    use once_cell::sync::Lazy;
    use regex::Regex;
    use serde::{Deserialize, Serialize};

    static OOM_RE: Lazy<Regex> = Lazy::new(|| {
        Regex::new(r"(?i)(OutOfMemoryError|java\.lang\.OutOfMemoryError|Killed|Cannot allocate memory|java heap space)")
            .expect("Invalid OOM regex")
    });

    static PLUGIN_RE: Lazy<Regex> = Lazy::new(|| {
        Regex::new(r"(?i)(NullPointerException|PluginClassLoader|java\.lang\.reflect\.InvocationTargetException|Caused by:.*Exception)")
            .expect("Invalid plugin exception regex")
    });

    #[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
    pub enum CrashType {
        #[serde(rename = "oom")]
        Oom,
        #[serde(rename = "config_error")]
        ConfigError,
        #[serde(rename = "plugin_crash")]
        PluginCrash,
        #[serde(rename = "generic")]
        Generic,
    }

    impl CrashType {
        pub fn as_str(&self) -> &'static str {
            match self {
                CrashType::Oom => "oom",
                CrashType::ConfigError => "config_error",
                CrashType::PluginCrash => "plugin_crash",
                CrashType::Generic => "generic",
            }
        }
    }

    /// Classify a crash based on container exit code and log excerpt.
    /// Pure function — no side effects, no DB access.
    pub fn classify_crash(exit_code: i32, log_excerpt: &str) -> CrashType {
        let has_oom = OOM_RE.is_match(log_excerpt);
        let has_plugin = PLUGIN_RE.is_match(log_excerpt);

        // Exit code 137 (SIGKILL) is strong OOM signal, cross-reference with log patterns
        if exit_code == 137 && has_oom {
            return CrashType::Oom;
        }

        // OOM can also happen with other exit codes if kernel detected memory pressure
        if has_oom {
            return CrashType::Oom;
        }

        // Plugin/mod crash: exception in plugin code (only if actually crashed, exit != 0)
        if has_plugin && exit_code != 0 {
            return CrashType::PluginCrash;
        }

        // Any other non-zero exit, no specific pattern detected
        CrashType::Generic
    }

    async fn handle_crash_report(&self, report: CrashReportData) {
        let server = match self.repository.find_by_id(&report.server_id).await {
            Ok(Some(s)) => s,
            Ok(None) => {
                tracing::error!("[CRASH] Crash report for unknown server: {}", report.server_id);
                return;
            }
            Err(e) => {
                tracing::error!("[CRASH] DB error fetching server {}: {}", report.server_id, e);
                return;
            }
        };

        let crash_type = crash_classifier::classify_crash(report.exit_code, &report.log_excerpt);

        // Crash-loop frequency check per D-03: if Generic or PluginCrash has 3+ rapid crashes
        // within 60s of startup, reclassify as ConfigError to disable auto-restart.
        // classify_crash() is a pure function with no DB access, so the frequency check
        // lives here in the service layer.
        let crash_type = match crash_type {
            CrashType::Generic | CrashType::PluginCrash => {
                let since = chrono::Utc::now() - chrono::Duration::from_secs(60);
                let recent = match self.get_crash_count(&server.id, since).await {
                    Ok(n) => n,
                    Err(_) => 0,
                };
                // +1 because current crash isn't stored yet
                if (recent + 1) >= 3 {
                    CrashType::ConfigError
                } else {
                    crash_type
                }
            }
            other => other,
        };

        match crash_type {
            CrashType::Oom => {
                // D-03: Notify only — do NOT auto-restart
                tracing::warn!("[CRASH] OOM detected for {} (exit={}) — notify only", server.name, report.exit_code);
                self.store_crash_log(&server, &report, &crash_type, "notified_only").await;
            }
            CrashType::ConfigError => {
                tracing::warn!("[CRASH] ConfigError crash-loop for {} — disabling auto_restart", server.name);
                if let Err(e) = self.disable_auto_restart(&server.id).await {
                    tracing::error!("[CRASH] Failed to disable auto_restart: {}", e);
                }
                self.store_crash_log(&server, &report, &crash_type, "restart_disabled").await;
            }
            CrashType::PluginCrash | CrashType::Generic => {
                // Follow Phase 57 auto-restart policy
                tracing::warn!("[CRASH] {:?} for {} — following auto-restart policy", crash_type, server.name);
                self.store_crash_log(&server, &report, &crash_type, "auto_restarted").await;
            }
        }

        // Notifications (D-04): toast + event timeline + Discord
        self.notify_crash(&server, &crash_type, &report).await;
    }
