pub async fn check_server_creation(
    &self,
    user_id: Uuid,
    requested_ram_mb: i64,
    requested_cpu: i64,
    requested_disk_gb: i64,
) -> Result<QuotaCheckResult, Box<dyn std::error::Error + Send + Sync>> {
    // Get plan limits (default to free if no subscription)
    let (max_servers, max_ram, max_cpu, max_disk) = self.get_plan_limits(user_id).await?;
    
    // Count existing servers
    let server_repo = SqlxServerRepository::new(self.pool.clone());
    let existing = server_repo.find_by_user_id(user_id).await?;
    let current_count = existing.len();
    let current_ram: i64 = existing.iter().map(|s| s.resources.get("ram").and_then(|r| r.as_str()).unwrap_or("0").parse::<i64>().unwrap_or(0)).sum();
    
    // Check limits
    let mut violations = Vec::new();
    
    if max_servers >= 0 && current_count >= max_servers {
        violations.push(QuotaViolation {
            resource: "servers".to_string(),
            requested: 1,
            limit: max_servers,
        });
    }
    // ... similar for RAM, CPU, disk
    
    Ok(QuotaCheckResult {
        allowed: violations.is_empty(),
        violations,
    })
}
