2. Add handler function:
```rust
pub async fn get_subscription(
    State(state): State<ApiState>,
    auth_user: AuthUser,
) -> Result<axum::response::Response, ApiError> {
    let sub_repo = SqlxSubscriptionRepository::new(state.pool.clone());
    let plan_repo = SqlxPlanRepository::new(state.pool.clone());
    
    let subscription = sub_repo.find_by_user_id(auth_user.user_id)
        .await
        .map_err(|e| ApiError::new("INTERNAL_ERROR", &e.to_string()))?;
    
    let response = if let Some(sub) = subscription {
        let plan = plan_repo.find_by_id(sub.plan_id)
            .await
            .map_err(|e| ApiError::new("INTERNAL_ERROR", &e.to_string()))?;
        
        serde_json::json!({
            "status": sub.status,
            "plan": plan.map(|p| serde_json::json!({
                "id": p.id,
                "name": p.name,
                "display_name": p.display_name,
                "limits": p.limits,
            })),
            "current_period_end": sub.current_period_end,
        })
    } else {
        // No subscription - return free tier
        serde_json::json!({
            "status": "none",
            "plan": null,
        })
    };
    
    Ok(ApiResponse::new(response).into_response())
}
```
