```jsx
function parseTemplateDefaults(template) {
  const config = template.config || {}
  const env = config.env || {}
  const ramMatch = (env.MEMORY || '').match(/(\d+)G/)
  return {
    name: template.display_name + ' Server',
    ram: ramMatch ? parseInt(ramMatch[1]) : 4,
    disk: 20,
    onlineMode: env.ONLINE_MODE !== 'false', // default true
    maxPlayers: parseInt(env.MAX_PLAYERS || '20'),
  }
}
```

```jsx
const loadNodes = async () => {
  setNodesLoading(true)

    const nodesList = await nodesApi.list()
```

**`loadNodeResources` function — fetch node capacity when node selected (D-07/D-09):**
```jsx
const loadNodeResources = async (nodeId) => {
  if (!nodeId) {
    setSelectedNodeResources(null)
    return
  }
  setResourcesLoading(true)
  try {
    const resources = await nodesApi.get(nodeId)
    setSelectedNodeResources(resources)
  } catch (err) {
    console.error('Failed to load node resources:', err)
    setSelectedNodeResources(null)
  } finally {
    setResourcesLoading(false)
  }
}
```
Note: `nodesApi.get(nodeId)` calls `GET /api/v1/nodes/:id` which returns node details including `total_memory_gb`. The exact endpoint is `GET /api/v1/nodes/:id/resources` according to RESEARCH.md. Use the existing `nodesApi.get(id)` which maps to `api.get('/nodes/${id}')`. If the response shape doesn't have `total_memory_gb`, fall back to checking `node.total_memory_gb` or skip the warning.

**`handleSubmit` function** (D-11 — submission is implicit confirmation, D-12):
```jsx
const handleSubmit = async (e) => {
  e.preventDefault()

  if (!serverName || serverName.trim() === '') {
    setValidationError('Server name is required')
    return
  }
  if (serverName.trim().length > 64) {
    setValidationError('Server name must be 64 characters or less')
    return
  }
  setValidationError('')
  setIsSubmitting(true)

  try {
    // Build payload matching CreateServerFromTemplateRequest DTO
    const payload = {
      name: serverName.trim(),
      node_id: nodeId || undefined,
      ram_mb: ram * 1024,
      max_players: parseInt(maxPlayers),
      minecraft_version: template.version || undefined,
      config_overrides: {
        online_mode: onlineMode === 'true',
        world_seed: worldSeed || undefined,
      },
    }

    const server = await templatesApi.createServer(template.id, payload)
    
    if (server && server.id) {
      addServer(server)
      const serversResponse = await serversApi.list()
      const allServers = serversResponse?.servers || serversResponse || []
      setServers(Array.isArray(allServers) ? allServers : [])
    }
    
    addToast({ type: 'success', message: 'Server created successfully!' })
    onClose()
    setTimeout(() => navigate(`/servers/${server.id}`), 500)
  } catch (err) {
    addToast({ type: 'error', message: err.message || 'Failed to create server' })
  } finally {
    setIsSubmitting(false)
  }
}
```

```jsx
const renderStepper = (value, setter, label, limits, step, unit) => {
  return (
    <div>
      <label className="text-sm text-[var(--color-text-secondary)] mb-1 block">{label}</label>
      <div className="flex items-center gap-3">
        <button
          onClick={() => setter(Math.max(limits.min, value - step))}
          disabled={value <= limits.min}
          className="w-8 h-8 rounded bg-[var(--color-bg-primary)] border border-[var(--color-cosmic-border)] text-[var(--color-text-primary)] hover:brightness-110 disabled:opacity-30 disabled:cursor-not-allowed"
          aria-label="Decrease"
        >-</button>
        <div className="flex-1 h-2 bg-[var(--color-cosmic-border)] rounded-full overflow-hidden">
          <div
            className="h-full bg-[var(--color-cosmic-cyan)] rounded-full transition-all"
            style={{ width: `${((value - limits.min) / (limits.max - limits.min)) * 100}%` }}
          />
        </div>
        <button
          onClick={() => setter(Math.min(limits.max, value + step))}
          disabled={value >= limits.max}
          className="w-8 h-8 rounded bg-[var(--color-bg-primary)] border border-[var(--color-cosmic-border)] text-[var(--color-text-primary)] hover:brightness-110 disabled:opacity-30 disabled:cursor-not-allowed"
          aria-label="Increase"
        >+</button>
        <span className="text-sm text-[var(--color-text-primary)] font-medium w-16 text-right">{value} {unit}</span>
      </div>
    </div>
  )
}
```
