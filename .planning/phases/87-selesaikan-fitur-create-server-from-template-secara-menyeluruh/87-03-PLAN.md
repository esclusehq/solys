**Change 2 — Add progress timer logic in handleSubmit:**

Modify the `handleSubmit` function. Before the try block, add progress timer setup:
```jsx
const handleSubmit = async (e) => {
  e.preventDefault()

  if (!serverName || serverName.trim() === '') {
    setValidationError('Server name is required')
    return
  }
  if (serverName.trim().length > 64) {
    setValidationError('Server name must be 64 characters or less')
    return
  }
  setValidationError('')
  setIsSubmitting(true)

  // Start progress timer (only shows if creation takes >2s)
  const startTime = Date.now()
  const steps = ['Creating server…', 'Installing dependencies…', 'Setting up configuration…']
  const checkInterval = setInterval(() => {
    const elapsed = Date.now() - startTime
    if (elapsed > 2000 && deployProgress === null) {
      setDeployProgress({ step: 0, steps, startTime })
    }
    // Advance step every 3 seconds
    if (elapsed > 2000) {
      const stepIndex = Math.min(Math.floor((elapsed - 2000) / 3000), steps.length - 1)
      setDeployProgress(prev => prev ? { ...prev, step: stepIndex } : { step: 0, steps, startTime })
    }
  }, 1000)
  progressCheckRef.current = checkInterval

  try {
    const payload = {
      name: serverName.trim(),
      node_id: nodeId || undefined,
      ram_mb: ram * 1024,
      max_players: parseInt(maxPlayers),
      minecraft_version: template.version || undefined,
      config_overrides: {
        online_mode: onlineMode === 'true',
        world_seed: worldSeed || undefined,
      },
    }

    const server = await templatesApi.createServer(template.id, payload)
    
    // Clear progress timer
    if (progressCheckRef.current) {
      clearInterval(progressCheckRef.current)
      progressCheckRef.current = null
    }
    setDeployProgress(null)

    if (server && server.id) {
      addServer(server)
      const serversResponse = await serversApi.list()
      const allServers = serversResponse?.servers || serversResponse || []
      setServers(Array.isArray(allServers) ? allServers : [])
    }
    
    addToast({ type: 'success', message: 'Server created successfully!' })
    onClose()
    setTimeout(() => navigate(`/servers/${server.id}`), 500)
  } catch (err) {
    // Clear progress timer
    if (progressCheckRef.current) {
      clearInterval(progressCheckRef.current)
      progressCheckRef.current = null
    }
    setDeployProgress(null)
    setIsSubmitting(false)
    addToast({ type: 'error', message: err.message || 'Failed to create server' })
  }
}
```
