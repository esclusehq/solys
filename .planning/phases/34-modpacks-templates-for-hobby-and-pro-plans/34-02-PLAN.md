  <action>
Create React hook for modpack templates:
- fetchModpackTemplates(gameType?) - fetches available modpacks
- Plan check: checks user plan from billing/subscription
- Only returns templates for Hobby/Pro plans, null/empty for Starter

```javascript
import { useState, useEffect } from 'react';
import { fetchModpackTemplates } from '../api/modpackTemplatesApi';
import { useServerStore } from '../store/serverStore';

export function useModpackTemplates() {
  const { user } = useServerStore(); // or useAuthStore
  
  // Plan check - Hobby/Pro/Enterprise only
  const isHobbyPlus = user?.subscription?.plan === 'hobby' || 
                      user?.subscription?.plan === 'pro' || 
                      user?.subscription?.plan === 'enterprise';
  
  const fetch = async (gameType) => {
    if (!isHobbyPlus) return []; // Starter can't use modpack templates
    return fetchModpackTemplates(gameType);
  };
  
  return { fetchModpackTemplates: fetch, isHobbyPlus };
}
```
