Implement D-22 (Graceful with timeout) in podman_server_executor.rs:

Replace the simple stop_server method with graceful stop that waits 30s:

```rust
async fn stop_server(&self, server: &Server) -> Result<()> {
    let container_name = self.get_container_name(server).await;
    
    // First, try graceful stop (SIGTERM)
    tracing::info!("Gracefully stopping container: {}", container_name);
    
    // Start container in background mode if not already
    let _ = self.run_podman(&["start", &container_name]).await;
    
    // Send SIGTERM (graceful stop)
    let term_result = self.run_podman(&["stop", "-t", "30", &container_name]).await;
    
    match term_result {
        Ok(_) => {
            tracing::info!("Container {} stopped gracefully", container_name);
            Ok(())
        }
        Err(e) => {
            // If graceful stop fails, check if container is still running
            let status = self.check_status(server).await?;
            if status == "running" {
                tracing::warn!("Graceful stop failed, forcing kill: {}", e);
                // Force kill as fallback
                self.run_podman(&["kill", "--signal", "KILL", &container_name]).await?;
                self.run_podman(&["rm", "-f", &container_name]).await?;
            }
            Ok(())
        }
    }
}
```

The -t 30 flag gives podman 30 seconds to stop gracefully before SIGKILL.
