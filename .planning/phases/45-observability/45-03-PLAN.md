<action>
Create trace ID middleware in web-agent/src/api/middleware/tracing.rs:

1. Define TraceId extractor that gets/sets X-Trace-ID header
2. Check for incoming X-Trace-ID header (from upstream proxy)
3. If not present, generate UUID v4 as trace_id
4. Inject trace_id into request extensions for use by handlers
5. Add response header X-Trace-ID with the trace_id value

```rust
use axum::{
    extract::Request,
    http::HeaderValue,
    middleware::Next,
};
use uuid::Uuid;

pub async fn trace_id_middleware(mut request: Request, next: Next) -> Response {
    // Get or create trace ID
    let trace_id = request
        .headers()
        .get("x-trace-id")
        .and_then(|v| v.to_str().ok())
        .map(String::from)
        .unwrap_or_else(|| Uuid::new_v4().to_string());
    
    // Inject into extensions
    request.extensions_mut().insert(trace_id.clone());
    
    // Call next handler
    let mut response = next.run(request).await;
    
    // Add trace ID to response headers
    response.headers_mut().insert(
        "x-trace-id",
        HeaderValue::from_str(&trace_id).unwrap(),
    );
    
    response
}
```
