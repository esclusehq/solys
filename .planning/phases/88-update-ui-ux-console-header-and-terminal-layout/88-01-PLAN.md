    import { useState, useRef, useEffect, useCallback } from 'react';
    import { Terminal } from '@xterm/xterm';
    import { FitAddon } from '@xterm/addon-fit';
    import { SearchAddon } from '@xterm/addon-search';
    import { WebLinksAddon } from '@xterm/addon-web-links';
    import '@xterm/xterm/css/xterm.css';
    import { TERMINAL_COSMIC_THEME, TERMINAL_FONT, TERMINAL_FONT_SIZE, TERMINAL_LINE_HEIGHT, TERMINAL_PROMPT } from '../theme/terminalTheme';
    import FloatingToolbar from './FloatingToolbar';

    export default function TerminalEmulator({ serverId, connected: externalConnected = false }) {
        const terminalRef = useRef(null);
        const terminalInstanceRef = useRef(null);
        const fitAddonRef = useRef(null);
        const searchAddonRef = useRef(null);

        const inputBuf = useRef('');
        const historyRef = useRef([]);
        const historyIdxRef = useRef(-1);
        const wsRef = useRef(null);
        const reconnectTimeoutRef = useRef(null);
        const reconnectAttempts = useRef(0);
        const searchInputRef = useRef(null);

        const [connected, setConnected] = useState(false);
        const [showSearch, setShowSearch] = useState(false);
        const [searchTerm, setSearchTerm] = useState('');

        const commandHistoryKey = `terminal_history_${serverId}`;

        // Load command history from localStorage
        useEffect(() => {
            try {
                const saved = localStorage.getItem(commandHistoryKey);
                if (saved) {
                    historyRef.current = JSON.parse(saved);
                }
            } catch (e) {
                console.error('[Terminal] Failed to load history:', e);
            }
        }, [commandHistoryKey]);

        // Save command history to localStorage
        const saveHistory = useCallback((newHistory) => {
            try {
                localStorage.setItem(commandHistoryKey, JSON.stringify(newHistory));
            } catch (e) {
                console.error('[Terminal] Failed to save history:', e);
            }
        }, [commandHistoryKey]);

        // Initialize terminal
        useEffect(() => {
            if (!terminalRef.current) return;

            const term = new Terminal({
                theme: TERMINAL_COSMIC_THEME,
                fontFamily: TERMINAL_FONT,
                fontSize: TERMINAL_FONT_SIZE,
                lineHeight: TERMINAL_LINE_HEIGHT,
                cursorBlink: true,
                cursorStyle: 'block',
                allowProposedSelection: true,
                scrollback: 10000,
            });

            const fitAddon = new FitAddon();
            term.loadAddon(fitAddon);

            const searchAddon = new SearchAddon({ highlightLimit: 1000 });
            term.loadAddon(searchAddon);
            searchAddonRef.current = searchAddon;

            const webLinksAddon = new WebLinksAddon();
            term.loadAddon(webLinksAddon);

            term.open(terminalRef.current);
            fitAddon.fit();

            terminalInstanceRef.current = term;
            fitAddonRef.current = fitAddon;

            // Write welcome message + prompt
            term.write('\r\n\x1b[36m░░░░░░░░░░░░░░░░░░░░░░░░░░░░\x1b[0m\r\n');
            term.write('\x1b[36m█░░ Terminal Ready ░░░░░░░░░\x1b[0m\r\n');
            term.write('\x1b[36m░░░░░░░░░░░░░░░░░░░░░░░░░░░░\x1b[0m\r\n\r\n');
            term.write('Connecting to server...\r\n');

            let currentBuffer = '';

            // Handle terminal input — inline buffer management
            term.onData((data) => {
                if (data === '\r') { // Enter
                    const cmd = currentBuffer.trim();
                    if (cmd) {
                        const newHistory = [...historyRef.current.slice(-49), cmd];
                        historyRef.current = newHistory;
                        historyIdxRef.current = -1;
                        saveHistory(newHistory);

                        term.write(`\r\n> ${cmd}\r\n`);

                        // Special commands
                        if (cmd.toLowerCase() === 'clear' || cmd.toLowerCase() === 'cls') {
                            term.clear();
                            term.write(`${TERMINAL_PROMPT}`);
                            currentBuffer = '';
                            return;
                        }

                        if (wsRef.current?.readyState === WebSocket.OPEN) {
                            wsRef.current.send(JSON.stringify({
                                type: 'command',
                                command: cmd,
                            }));
                        } else {
                            term.write('\x1b[31m✗ Not connected\x1b[0m\r\n');
                        }
                    }
                    term.write(`${TERMINAL_PROMPT}`);
                    currentBuffer = '';
                } else if (data === '\x1b[A') { // Arrow Up
                    if (historyRef.current.length === 0) return;
                    const idx = historyIdxRef.current === -1
                        ? historyRef.current.length - 1
                        : Math.max(historyIdxRef.current - 1, 0);
                    historyIdxRef.current = idx;
                    const line = historyRef.current[idx];
                    term.write('\r\x1b[K');
                    term.write(`${TERMINAL_PROMPT}${line}`);
                    currentBuffer = line;
                } else if (data === '\x1b[B') { // Arrow Down
                    if (historyIdxRef.current === -1) return;
                    const idx = historyIdxRef.current + 1;
                    if (idx >= historyRef.current.length) {
                        historyIdxRef.current = -1;
                        term.write(`\r\x1b[K${TERMINAL_PROMPT}`);
                        currentBuffer = '';
                    } else {
                        historyIdxRef.current = idx;
                        const line = historyRef.current[idx];
                        term.write(`\r\x1b[K${TERMINAL_PROMPT}${line}`);
                        currentBuffer = line;
                    }
                } else if (data === '\x1b[D' || data === '\x1b[C') { // Arrow left/right — no-op
                    // pass — cursor movement handled intrinsically by xterm
                } else if (data === '\x7f' || data === '\x08') { // Backspace
                    if (currentBuffer.length > 0) {
                        currentBuffer = currentBuffer.slice(0, -1);
                        term.write('\b \b');
                    }
                } else if (data === '\x03') { // Ctrl+C
                    term.write('^C\r\n');
                    term.write(TERMINAL_PROMPT);
                    currentBuffer = '';
                } else if (data === '\x0c') { // Ctrl+L — clear
                    term.clear();
                    term.write(`${TERMINAL_PROMPT}${currentBuffer}`);
                } else if (data === '\x06') { // Ctrl+F — search toggle
                    setShowSearch(prev => !prev);
                } else if (data >= ' ' || data === '\t') { // Printable + Tab
                    currentBuffer += data;
                    term.write(data);
                }
            });

            // Fit on resize
            const handleResize = () => {
                fitAddon.fit();
            };
            window.addEventListener('resize', handleResize);

            // Use ResizeObserver for more reliable sizing (sidebar collapse etc.)
            const resizeObserver = new ResizeObserver(() => {
                fitAddon.fit();
            });
            if (terminalRef.current?.parentElement) {
                resizeObserver.observe(terminalRef.current.parentElement);
            }

            setTimeout(() => fitAddon.fit(), 100);

            return () => {
                window.removeEventListener('resize', handleResize);
                resizeObserver.disconnect();
                term.dispose();
            };
        }, [saveHistory]);

        // Connect to WebSocket
        useEffect(() => {
            if (!serverId) return;

            const connect = () => {
                const wsUrl = `/ws/terminal/${serverId}`;
                const ws = new WebSocket(wsUrl);
                wsRef.current = ws;

                ws.onopen = () => {
                    setConnected(true);
                    reconnectAttempts.current = 0;
                    terminalInstanceRef.current?.write('\r\n\x1b[32m✓ Connected\x1b[0m\r\n');
                    terminalInstanceRef.current?.write(`${TERMINAL_PROMPT}`);
                };

                ws.onmessage = (event) => {
                    try {
                        const msg = JSON.parse(event.data);
                        if (msg.type === 'output') {
                            const text = msg.data?.text || msg.data?.line || msg.data || '';
                            terminalInstanceRef.current?.write(text);
                        } else if (msg.type === 'error') {
                            terminalInstanceRef.current?.write(`\r\n\x1b[31mError: ${msg.message}\x1b[0m\r\n`);
                        }
                    } catch (e) {
                        terminalInstanceRef.current?.write(event.data);
                    }
                };

                ws.onclose = () => {
                    setConnected(false);
                    terminalInstanceRef.current?.write('\r\n');
                    if (reconnectAttempts.current < 10) {
                        const delay = Math.min(1000 * Math.pow(2, reconnectAttempts.current), 30000);
                        reconnectTimeoutRef.current = setTimeout(() => {
                            reconnectAttempts.current++;
                            connect();
                        }, delay);
                    }
                };

                ws.onerror = (err) => {
                    console.error('[Terminal] WebSocket error:', err);
                };
            };

            connect();

            return () => {
                if (reconnectTimeoutRef.current) {
                    clearTimeout(reconnectTimeoutRef.current);
                }
                if (wsRef.current) {
                    wsRef.current.close();
                }
            };
        }, [serverId]);

        const reconnect = useCallback(() => {
            if (reconnectTimeoutRef.current) {
                clearTimeout(reconnectTimeoutRef.current);
            }
            if (wsRef.current) {
                wsRef.current.close();
            }
            reconnectAttempts.current = 0;
            const connect = () => {
                const wsUrl = `/ws/terminal/${serverId}`;
                const ws = new WebSocket(wsUrl);
                wsRef.current = ws;
                ws.onopen = () => {
                    setConnected(true);
                    reconnectAttempts.current = 0;
                    terminalInstanceRef.current?.write('\r\n\x1b[32m✓ Connected\x1b[0m\r\n');
                    terminalInstanceRef.current?.write(`${TERMINAL_PROMPT}`);
                };
                ws.onmessage = (event) => {
                    try {
                        const msg = JSON.parse(event.data);
                        if (msg.type === 'output') {
                            terminalInstanceRef.current?.write(msg.data?.text || msg.data?.line || msg.data || '');
                        } else if (msg.type === 'error') {
                            terminalInstanceRef.current?.write(`\r\n\x1b[31mError: ${msg.message}\x1b[0m\r\n`);
                        }
                    } catch (e) {

                            connect();

            }
        }, []);

            setShowSearch(prev => !prev);
            if (!showSearch) {
                setTimeout(() => searchInputRef.current?.focus(), 0);
            } else {
                searchAddonRef.current?.clearDecorations();
                setSearchTerm('');
            }
        }, [showSearch]);

        const handleSearchInput = useCallback((e) => {
            const term = e.target.value;
            setSearchTerm(term);
            if (term) {
                searchAddonRef.current?.findNext(term, {
                    decorations: {
                        matchOverviewRuler: '#0ddff2',
                        matchBackground: '#0ddff220',
                        activeMatchBackground: '#0ddff240',
                        activeMatchColorOverviewRuler: '#0ddff2',
                    },
                    incremental: true,
                });
            } else {
                searchAddonRef.current?.clearDecorations();
            }
        }, []);

            if (searchTerm) {
                searchAddonRef.current?.findPrevious(searchTerm);
            }
        }, [searchTerm]);

            if (searchTerm) {
                searchAddonRef.current?.findNext(searchTerm);
            }
        }, [searchTerm]);

        const handleSearchClose = useCallback(() => {
            setShowSearch(false);
            setSearchTerm('');
            searchAddonRef.current?.clearDecorations();
            terminalInstanceRef.current?.focus();
        }, []);

        // Reconnect overlay countdown
        const [reconnectCountdown, setReconnectCountdown] = useState(0);
        useEffect(() => {
            if (connected || reconnectAttempts.current === 0) {
                setReconnectCountdown(0);
                return;
            }
            const delay = Math.min(1000 * Math.pow(2, reconnectAttempts.current - 1), 30000);
            const seconds = Math.ceil(delay / 1000);
            setReconnectCountdown(seconds);
            const interval = setInterval(() => {
                setReconnectCountdown(prev => Math.max(0, prev - 1));
            }, 1000);
            return () => clearInterval(interval);
        }, [connected, reconnectAttempts.current]);

        const isDisconnected = !connected && reconnectAttempts.current > 0;
        const maxAttempts = 10;

                {/* Terminal Title Bar — status indicator */}
                <div className="shrink-0 flex items-center justify-between px-2 py-1 border-b border-[var(--color-cosmic-border)]">
                    <div className="flex items-center gap-2">
                        <span className={`w-2 h-2 rounded-full ${
                            connected
                                ? 'bg-[var(--color-cosmic-green)] shadow-[0_0_6px_var(--color-cosmic-green)]'
                                : 'bg-[var(--color-cosmic-red)]'
                        }`} />
                        <span className={`text-xs font-semibold ${
                            connected
                                ? 'text-[var(--color-cosmic-green)]'
                                : 'text-[var(--color-cosmic-red)]'
                        }`}>
                            {connected ? 'Connected' : 'Disconnected'}
                        </span>
                    </div>
                </div>

                {/* Terminal Container — fills remaining height */}
                <div className="flex-1 relative overflow-hidden glass-panel border-x border-[var(--color-cosmic-border)]">
                    <div
                        ref={terminalRef}
                        className="h-full w-full p-0"
                        style={{ minHeight: '300px' }}
                    />

                        onClear={handleClear}
                        onReconnect={reconnect}
                        onSearchToggle={handleSearchToggle}
                        onCopy={handleCopy}
                        isConnected={connected}
                    />

                    {/* Reconnect Overlay — translucent overlay on disconnect */}
                    {isDisconnected && (
                        <div className="absolute inset-0 z-20 flex items-center justify-center">
                            <div className="absolute inset-0 bg-[var(--color-deep-space)]/60 backdrop-blur-[2px] pointer-events-none" />
                            <div className="relative z-10 text-center pointer-events-auto">
                                {atMaxAttempts ? (
                                    <>
                                        <p className="text-[var(--color-cosmic-red)] text-sm font-semibold mb-2">
                                            Disconnected
                                        </p>
                                        <p className="text-[var(--color-text-muted)] text-xs mb-4">
                                            Max reconnection attempts reached.
                                        </p>
                                    </>
                                ) : (
                                    <>
                                        <p className="text-[var(--color-cosmic-red)] text-sm font-semibold mb-2">
                                            Disconnected
                                        </p>
                                        <p className="text-[var(--color-text-muted)] text-xs mb-4">
                                            Reconnecting in {reconnectCountdown}s... (attempt {reconnectAttempts.current}/{maxAttempts})
                                        </p>
                                    </>
                                )}
                                <button
                                    onClick={reconnect}
                                    className="px-4 py-2 rounded-lg text-xs font-semibold
                                               bg-[var(--color-cosmic-cyan)]/10 text-[var(--color-cosmic-cyan)]
                                               border border-[var(--color-cosmic-cyan)]/30
                                               hover:bg-[var(--color-cosmic-cyan)]/20 transition-all"
                                >
                                    Retry Now
                                </button>
                            </div>
                        </div>
                    )}

                    {/* Search Bar — Ctrl+F overlay */}
                    {showSearch && (
                        <div className="absolute top-10 right-2 z-10 flex items-center gap-2
                                        bg-[var(--color-cosmic-card)] border border-[var(--color-cosmic-border)]

                                type="text"
                                value={searchTerm}
