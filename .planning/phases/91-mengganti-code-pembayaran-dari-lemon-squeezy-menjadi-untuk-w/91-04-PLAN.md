                             Set LEMON_SQUEEZY_WEBHOOK_SECRET environment variable for webhook verification."

                        tracing::warn!(
                            "LEMON_SQUEEZY_API_KEY is not set - payment features will be disabled"

                    }
                },
                _ => anyhow::bail!("Unknown BILLING_PROVIDER: {}. Must be 'whop' or 'lemon_squeezy'.", self.billing_provider),
