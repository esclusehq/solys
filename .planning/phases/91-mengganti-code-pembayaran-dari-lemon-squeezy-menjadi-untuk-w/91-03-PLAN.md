/// D-21: Type-safe subscription status with is_active() entitlement check.
/// Maps from Whop membership statuses and persists as lowercase string.
#[derive(Debug, Clone, Serialize, Deserialize, PartialEq)]
pub enum SubscriptionStatus {
    Trialing,
    Active,
    PastDue,
    Canceled,
    Expired,
}

impl SubscriptionStatus {
    pub fn is_active(&self) -> bool {
        matches!(self, SubscriptionStatus::Active | SubscriptionStatus::Trialing)
    }
}

impl fmt::Display for SubscriptionStatus {
    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
        match self {
            SubscriptionStatus::Trialing => write!(f, "trialing"),
            SubscriptionStatus::Active => write!(f, "active"),
            SubscriptionStatus::PastDue => write!(f, "past_due"),
            SubscriptionStatus::Canceled => write!(f, "canceled"),
            SubscriptionStatus::Expired => write!(f, "expired"),
        }
    }
}

impl FromStr for SubscriptionStatus {
    type Err = String;
    fn from_str(s: &str) -> Result<Self, Self::Err> {
        match s {
            "trialing" => Ok(SubscriptionStatus::Trialing),
            "active" => Ok(SubscriptionStatus::Active),
            "past_due" => Ok(SubscriptionStatus::PastDue),
            "canceled" | "cancelled" => Ok(SubscriptionStatus::Canceled),
            "expired" | "completed" => Ok(SubscriptionStatus::Expired),
            "unresolved" | "unpaid" => Ok(SubscriptionStatus::PastDue),
            _ => Err(format!("Unknown subscription status: {}", s)),
        }
    }
}

   /// D-06: Dedicated Whop webhook endpoint with Standard Webhooks verification (D-08)
   pub async fn whop_webhook(
       State(state): State<ApiState>,
       headers: HeaderMap,
       body: String,
   ) -> Result<axum::response::Response, ApiError> {
       let secret = state.billing_webhook_secret.clone()
           .ok_or_else(|| ApiError::new("WEBHOOK_NOT_CONFIGURED", "Whop webhook secret not configured"))?;

       // 1. Extract Standard Webhooks headers
       let webhook_id = headers.get("webhook-id")
           .and_then(|v| v.to_str().ok())
           .ok_or_else(|| ApiError::new("MISSING_HEADER", "Missing webhook-id header"))?;

       let webhook_timestamp = headers.get("webhook-timestamp")
           .and_then(|v| v.to_str().ok())
           .ok_or_else(|| ApiError::new("MISSING_HEADER", "Missing webhook-timestamp header"))?;

       let webhook_signature = headers.get("webhook-signature")
           .and_then(|v| v.to_str().ok())
           .ok_or_else(|| ApiError::new("MISSING_HEADER", "Missing webhook-signature header"))?;

       // 2. Verify timestamp (±5 minutes tolerance per D-08)
       let timestamp: i64 = webhook_timestamp.parse()
           .map_err(|_| ApiError::new("INVALID_TIMESTAMP", "Invalid webhook timestamp"))?;
       let now = chrono::Utc::now().timestamp();
       if (now - timestamp).abs() > 300 {
           return Err(ApiError::new("STALE_WEBHOOK", "Webhook timestamp outside ±5 minute tolerance"));
       }

       // 3. Verify signature (Standard Webhooks HMAC-SHA256)
       let signed_content = format!("{}.{}.{}", webhook_id, webhook_timestamp, body);

       let secret_bytes = base64::engine::general_purpose::STANDARD
           .decode(&secret)
           .map_err(|_| ApiError::new("INVALID_SECRET", "Invalid webhook secret encoding"))?;

       let mut mac = hmac::Hmac::<sha2::Sha256>::new_from_slice(&secret_bytes)
           .map_err(|_| ApiError::new("HMAC_ERROR", "Failed to create HMAC"))?;
       mac.update(signed_content.as_bytes());
       let result = mac.finalize();
       let computed_sig = base64::engine::general_purpose::STANDARD.encode(result.into_bytes());

       let sig_value = webhook_signature
           .strip_prefix("v1,")
           .ok_or_else(|| ApiError::new("INVALID_SIGNATURE", "Invalid signature format"))?;

       use subtle::ConstantTimeEq;
       if !bool::from(computed_sig.as_bytes().ct_eq(sig_value.as_bytes())) {
           return Err(ApiError::new("INVALID_SIGNATURE", "Webhook signature verification failed"));
       }

       // 4. Check dedup (processed_webhook_events table)
       let existing: Option<(String,)> = sqlx::query_scalar(
           "SELECT event_id FROM processed_webhook_events WHERE provider = 'whop' AND event_id = $1"
       )
       .bind(webhook_id)
       .fetch_optional(&state.pool)
       .await
       .map_err(|e| ApiError::new("DB_ERROR", &e.to_string()))?;

       if existing.is_some() {
           tracing::debug!("Duplicate Whop webhook {} skipped", webhook_id);
           return Ok(ApiResponse::new(serde_json::json!({ "received": true, "duplicate": true })).into_response());
       }

       // 5. Log to billing_events (D-09)
       let event_type_value = serde_json::from_str::<serde_json::Value>(&body)
           .ok()
           .and_then(|v| v.get("type").and_then(|t| t.as_str()))
           .unwrap_or("unknown")
           .to_string();

       let headers_json = serde_json::json!({
           "webhook-id": webhook_id,
           "webhook-timestamp": webhook_timestamp,
           "webhook-signature": webhook_signature,
       });

       sqlx::query(
           "INSERT INTO billing_events (provider, event_type, event_id, raw_payload, raw_headers, processing_status) VALUES ($1, $2, $3, $4::jsonb, $5::jsonb, 'pending')"
       )
       .bind("whop")
       .bind(&event_type_value)
       .bind(webhook_id)
       .bind(&body)
       .bind(&headers_json.to_string())
       .execute(&state.pool)
       .await
       .map_err(|e| ApiError::new("DB_ERROR", &e.to_string()))?;

       // 6. Parse and process event
       let event: serde_json::Value = serde_json::from_str(&body)
           .map_err(|e| ApiError::new("INVALID_PAYLOAD", &e.to_string()))?;

       let event_type = event.get("type")
           .and_then(|v| v.as_str())
           .unwrap_or("unknown");

       tracing::info!("Processing Whop webhook: {}", event_type);

       let billing_service = crate::domain::billing::service::BillingService::new(state.pool.clone());

       // 7. Route event type to handler
       match event_type {
           "payment.succeeded" => {
               if let Some(data) = event.get("data") {
                   let user_id = data.get("metadata")
                       .and_then(|m| m.get("user_id"))
                       .and_then(|u| u.as_str())
                       .and_then(|s| Uuid::parse_str(s).ok())
                       .unwrap_or(Uuid::nil());

                   let member_id = data.get("member")
                       .and_then(|m| m.get("id"))
                       .and_then(|id| id.as_str());

                   let plan_id = event.get("data")
                       .and_then(|d| d.get("metadata"))
                       .and_then(|m| m.get("plan_id"))
                       .and_then(|p| p.as_str())
                       .and_then(|s| Uuid::parse_str(s).ok())
