```rust
/// Get log directory path with fallbacks
/// Primary: /var/log/escluse-agent/ (D-06)
/// Fallback: ~/.local/share/escluse-agent/logs/ (D-07)
/// Last fallback: stdout (D-08 - for containerized environments)
pub fn get_log_dir() -> Option<tracing_appender::non_blocking::WorkerGuard> {
    // Try primary location /var/log/escluse-agent/
    let primary_path = PathBuf::from("/var/log/escluse-agent");
    
    if primary_path.exists() || std::fs::create_dir_all(&primary_path).is_ok() {
        // Create appender with rotation
        let file_appender = tracing_appender::rolling::builder()
            .rotation(tracing_appender::rotation::HOURLY)  // Rotate hourly (checked daily limit also)
            .max_log_files(5)  // D-09: keep 5 files
            .filename_prefix("agent")
            .filename_suffix("log")
            .build(&primary_path)
            .ok()?;
            
        return Some(tracing_appender::non_blocking(file_appender));
    }
    
    // Try fallback: ~/.local/share/escluse-agent/logs/
    if let Some(local_dir) = dirs::data_local_dir() {
        let fallback_path = local_dir.join("escluse-agent").join("logs");
        if fallback_path.exists() || std::fs::create_dir_all(&fallback_path).is_ok() {
            let file_appender = tracing_appender::rolling::builder()
                .rotation(tracing_appender::rotation::HOURLY)
                .max_log_files(5)
                .filename_prefix("agent")
                .filename_suffix("log")
                .build(&fallback_path)
                .ok()?;
                
            return Some(tracing_appender::non_blocking(file_appender));
        }
    }
    
    // D-08: No file logging - return None for stdout fallback
    None
}
```
