    ```rust
    use std::sync::Arc;
    use std::time::Duration;
    use dashmap::DashMap;
    use tokio::net::UdpSocket;
    use tracing::{info, warn};
    use uuid::Uuid;

    /// Per-port state tracked by UdpPortRegistry.
    pub struct UdpPortEntry {
        pub server_id: Uuid,
        pub socket: Arc<UdpSocket>,
        pub grace_task: Option<tokio::task::JoinHandle<()>>,
        /// Task handle for the per-port recv_from forwarding loop.
        /// Spawned after the TunnelHandle is registered in the Registry.
        pub session_task: Option<tokio::task::JoinHandle<()>>,
    }

    /// Thread-safe registry mapping UDP port → server_id + UdpSocket.
    /// Used by the gateway to route incoming Bedrock player datagrams
    /// to the correct agent tunnel.
    pub struct UdpPortRegistry {
        ports: Arc<DashMap<u16, UdpPortEntry>>,
    }

    impl UdpPortRegistry {
        pub fn new() -> Self {
            Self {
                ports: Arc::new(DashMap::new()),
            }
        }

        /// Bind a UdpSocket on `port` for `server_id`.
        /// Called when TunnelConnect with loader="bedrock" is received (D-03/D-04).
        /// Returns Err if the port is already bound.
        pub async fn bind_port(
            &self,
            port: u16,
            server_id: Uuid,
        ) -> Result<(), std::io::Error> {
            if self.ports.contains_key(&port) {
                return Err(std::io::Error::new(
                    std::io::ErrorKind::AddrInUse,
                    format!("UDP port {} already bound", port),
                ));
            }
            let socket = UdpSocket::bind(format!("0.0.0.0:{}", port)).await?;
            self.ports.insert(port, UdpPortEntry {
                server_id,
                socket: Arc::new(socket),
                grace_task: None,
                session_task: None,
            });
            info!("[UDP] Bound port {} for server_id={}", port, server_id);
            Ok(())
        }

        /// Get the UdpPortEntry for a port, if bound.
        pub fn get_port(&self, port: u16) -> Option<dashmap::mapref::one::Ref<u16, UdpPortEntry>> {
            self.ports.get(&port)
        }

        /// Look up server_id by port.
        pub fn lookup_server_id(&self, port: u16) -> Option<Uuid> {
            self.ports.get(&port).map(|e| e.server_id)
        }

        /// Start the 30-second grace period (D-05). After `duration`, the port
        /// entry is removed and the socket is dropped (closed).
        /// If a grace task is already running for this port, it is aborted first
        /// (re-entering grace on re-disconnect resets the timer).
        pub async fn start_grace_period(&self, port: u16, duration: Duration) {
            if let Some(mut entry) = self.ports.get_mut(&port) {
                // Cancel any existing grace task
                if let Some(task) = entry.grace_task.take() {
                    task.abort();
                }
                let ports = self.ports.clone();
                let p = port;
                entry.grace_task = Some(tokio::spawn(async move {
                    tokio::time::sleep(duration).await;
                    ports.remove(&p);
                    info!("[UDP] Grace period expired, freed port {}", p);
                }));
            }
        }

        /// Cancel the grace period (called on reconnect).
        pub fn cancel_grace(&self, port: u16) {
            if let Some(mut entry) = self.ports.get_mut(&port) {
                if let Some(task) = entry.grace_task.take() {
                    task.abort();
                    info!("[UDP] Grace period cancelled for port {}", port);
                }
            }
        }

        /// Immediately free a port (no grace). Used when the tunnel
        /// disconnect reason implies the server is deleted.
        pub fn free_port(&self, port: u16) {
            if let Some(entry) = self.ports.get(&port) {
                if let Some(task) = &entry.grace_task {
                    task.abort();
                }
            }
            self.ports.remove(&port);
            info!("[UDP] Freed port {} (immediate)", port);
        }

        /// Check if a port is currently bound.
        pub fn is_bound(&self, port: u16) -> bool {
            self.ports.contains_key(&port)
        }

        /// Return number of currently bound UDP ports.
        pub fn len(&self) -> usize {
            self.ports.len()
        }

        /// Spawn a per-port recv_from forwarding session (D-04, D-07).
        /// Opens a yamux stream to the agent on first datagram and
        /// forwards datagrams bidirectionally with TLV framing.
        /// The `yamux_control` is cloned from the TunnelHandle.
        pub fn spawn_session(
            &self,
            port: u16,
            yamux_control: tokio_yamux::Control,
        ) -> bool {
            let entry = match self.ports.get(&port) {
                Some(e) => e.clone(),
                None => return false,
            };
            if entry.session_task.is_some() {
                return true; // already running
            }
            let ports = self.ports.clone();
            let socket = entry.socket.clone();
            let server_id = entry.server_id;
            let handle = tokio::spawn(async move {
                run_udp_player_session(socket, server_id, yamux_control, ports, port).await;
            });
            if let Some(mut entry) = self.ports.get_mut(&port) {
                entry.session_task = Some(handle);
            }
            true
        }

        /// Abort the session task for a port (called on disconnect/cleanup).
        pub fn abort_session(&self, port: u16) {
            if let Some(mut entry) = self.ports.get_mut(&port) {
                if let Some(task) = entry.session_task.take() {
                    task.abort();
                }
            }
        }
    }

    /// Gateway UDP player session: reads datagrams from a bound UdpSocket,
    /// opens a yamux stream to the agent on first datagram, and forwards
    /// datagrams bidirectionally with TLV framing (D-07, D-08).
    ///
    /// This function runs as a per-port tokio task. It is spawned when
    /// the TunnelHandle is registered (after bind_port).
    pub async fn run_udp_player_session(
        socket: Arc<UdpSocket>,
        server_id: Uuid,
        yamux_control: tokio_yamux::Control,
        ports: Arc<DashMap<u16, UdpPortEntry>>,
        port: u16,
    ) {
        let mut buf = vec![0u8; 65535];
        let mut session_started = false;
        let mut yamux_read: Option<tokio::io::ReadHalf<tokio_yamux::StreamHandle>> = None;
        let mut yamux_write: Option<tokio::io::WriteHalf<tokio_yamux::StreamHandle>> = None;
        let mut recv_buf = vec![0u8; 65535];

        // Ensure the port is still bound before starting
        if !ports.contains_key(&port) {
            return;
        }

        loop {
            // Check if port is still bound (grace period may have freed it)
            if !ports.contains_key(&port) {
                info!("[UDP] Port {} freed — stopping player session", port);
                break;
            }

            if !session_started {
                // Wait for first datagram (D-04 deferred session)
                let (n, src) = match socket.recv_from(&mut buf).await {
                    Ok(v) => v,
                    Err(e) => {
                        warn!("[UDP] recv_from error on port {}: {}", port, e);
                        break;
                    }
                };

                // Open yamux stream to agent on first datagram
                let mut stream = match yamux_control.open_stream().await {
                    Ok(s) => s,
                    Err(e) => {
                        warn!("[UDP] Failed to open yamux stream for port {}: {}", port, e);
                        // Keep trying — next datagram will retry
                        continue;
                    }
                };

                let (r, w) = tokio::io::split(stream);
                yamux_read = Some(r);
                yamux_write = Some(w);
                session_started = true;

                // Forward the first datagram immediately (TLV encode)
                let write = yamux_write.as_mut().unwrap();
                if let Err(e) = write.write_all(&[0x01]).await {
                    warn!("[UDP] Failed to write TLV type: {}", e); break;
                }
                if let Err(e) = write.write_all(&(n as u32).to_be_bytes()).await {
                    warn!("[UDP] Failed to write TLV length: {}", e); break;
                }
                if let Err(e) = write.write_all(&buf[..n]).await {
                    warn!("[UDP] Failed to write datagram: {}", e); break;
                }
                if let Err(e) = write.flush().await {
                    warn!("[UDP] Flush failed: {}", e); break;
                }
                info!("[UDP] First datagram forwarded: port={}, {} bytes from {}", port, n, src);

                // Spawn the reverse task (yamux→UdpSocket) for this session.
                // ReadHalf goes to the spawn task; WriteHalf stays in the main loop.
                let socket_clone = socket.clone();
                let mut yamux_read_task = yamux_read.take().unwrap();
                tokio::spawn(async move {
                    let mut len_buf = [0u8; 4];
                    let mut payload = vec![0u8; 65535];
                    loop {
                        let mut type_byte = [0u8; 1];
                        if yamux_read_task.read_exact(&mut type_byte).await.is_err() { break; }
                        if type_byte[0] != 0x01 { continue; }
                        if yamux_read_task.read_exact(&mut len_buf).await.is_err() { break; }
                        let len = u32::from_be_bytes(len_buf) as usize;
                        if len > payload.len() { break; }
                        if yamux_read_task.read_exact(&mut payload[..len]).await.is_err() { break; }
                        if socket_clone.send_to(&payload[..len], src).await.is_err() { break; }
                    }
                });
            } else {
                // Session already established — forward datagrams
                let (n, _src) = match socket.recv_from(&mut recv_buf).await {
                    Ok(v) => v,
                    Err(e) => {
                        warn!("[UDP] recv_from error on port {}: {}", port, e);
                        break;
                    }
                };
                let write = yamux_write.as_mut().unwrap();
                if write.write_all(&[0x01]).await.is_err() { break; }
                if write.write_all(&(n as u32).to_be_bytes()).await.is_err() { break; }
                if write.write_all(&recv_buf[..n]).await.is_err() { break; }
                if write.flush().await.is_err() { break; }
            }
        }
    }
    ```
