    ```rust
    /// Create a Route 53 SRV record for Bedrock player discovery (D-13, D-14).
    ///
    /// Format: `_minecraft._udp.bedrock-{subdomain}.play.esluce.com` →
    /// `0 0 {port} relay.esluce.net`
    ///
    /// Uses environment variables for AWS config:
    /// - `ROUTE53_HOSTED_ZONE_ID` — Route 53 hosted zone ID for play.esluce.com
    /// - `AWS_ACCESS_KEY_ID` / `AWS_SECRET_ACCESS_KEY` — AWS credentials
    ///
    /// This is a best-effort operation — failures are logged but not propagated
    /// as errors (DNS should not block server creation).
    pub async fn create_srv_record(&self, subdomain: &str, port: u16) {
        let hosted_zone_id = match std::env::var("ROUTE53_HOSTED_ZONE_ID") {
            Ok(id) => id,
            Err(_) => {
                tracing::warn!("[SRV] ROUTE53_HOSTED_ZONE_ID not set, skipping SRV record creation");
                return;
            }
        };

        let record_name = format!("_minecraft._udp.bedrock-{}.play.esluce.com", subdomain);
        let record_value = format!("0 0 {} relay.esluce.net", port);

        tracing::info!(
            "[SRV] Creating SRV record: {} → {}",
            record_name,
            record_value
        );

        // Use the aws-sdk-route53 client if available, otherwise log a warning.
        // The AWS SDK is not yet a dependency — we add it as optional and
        // gracefully degrade if it's not available.
        #[cfg(feature = "route53")]
        {
            use aws_config::meta::region::RegionProviderChain;
            use aws_sdk_route53::types::{
                Change, ChangeAction, ResourceRecord, ResourceRecordSet, RrType,
            };

            let region_provider = RegionProviderChain::default_provider()
                .or_else("us-east-1");
            let config = aws_config::from_env()
                .region(region_provider)
                .load()
                .await;
            let client = aws_sdk_route53::Client::new(&config);

            let change = Change::builder()
                .action(ChangeAction::Upsert)
                .resource_record_set(
                    ResourceRecordSet::builder()
                        .name(record_name.clone())
                        .type_(RrType::Srv)
                        .ttl(60)
                        .resource_records(
                            ResourceRecord::builder()
                                .value(record_value.clone())
                                .build()
                        )
                        .build()
                )
                .build();

            match client
                .change_resource_record_sets()
                .hosted_zone_id(&hosted_zone_id)
                .change_batch(
                    aws_sdk_route53::types::ChangeBatch::builder()
                        .changes(change)
                        .build()
                )
                .send()
                .await
            {
                Ok(_) => tracing::info!("[SRV] SRV record created successfully: {}", record_name),
                Err(e) => tracing::warn!("[SRV] Failed to create SRV record: {}", e),
            }
        }

        #[cfg(not(feature = "route53"))]
        {
            tracing::warn!(
                "[SRV] aws-sdk-route53 not available (feature 'route53' not enabled). \
                 Would create: {} → {}",
                record_name,
                record_value
            );
        }
    }
    ```

    ```rust
    /// Delete a Route 53 SRV record (called when Bedrock server is deleted).
    pub async fn delete_srv_record(&self, subdomain: &str) {
        let hosted_zone_id = match std::env::var("ROUTE53_HOSTED_ZONE_ID") {
            Ok(id) => id,
            Err(_) => {
                tracing::warn!("[SRV] ROUTE53_HOSTED_ZONE_ID not set, skipping SRV record deletion");
                return;
            }
        };

        let record_name = format!("_minecraft._udp.bedrock-{}.play.esluce.com", subdomain);

        #[cfg(feature = "route53")]
        {
            use aws_config::meta::region::RegionProviderChain;
            use aws_sdk_route53::types::{
                Change, ChangeAction, ResourceRecord, ResourceRecordSet, RrType,
            };

            let region_provider = RegionProviderChain::default_provider()
                .or_else("us-east-1");
            let config = aws_config::from_env()
                .region(region_provider)
                .load()
                .await;
            let client = aws_sdk_route53::Client::new(&config);

            // For deletion, we need the current record value. We query first,
            // then delete with the exact value. Since this is best-effort,
            // we log failures and continue.
            let list_result = client
                .list_resource_record_sets()
                .hosted_zone_id(&hosted_zone_id)
                .set_name(Some(record_name.clone()))
                .set_type(Some(RrType::Srv))
                .send()
                .await;

            match list_result {
                Ok(list_output) => {
                    if let Some(sets) = list_output.resource_record_sets {
                        for set in sets {
                            if set.name.trim_end_matches('.') == record_name.trim_end_matches('.')
                                && set.type_ == RrType::Srv
                            {
                                if let Some(records) = set.resource_records {
                                    let change = Change::builder()
                                        .action(ChangeAction::Delete)
                                        .resource_record_set(
                                            ResourceRecordSet::builder()
                                                .name(record_name.clone())
                                                .type_(RrType::Srv)
                                                .ttl(set.ttl.unwrap_or(60))
                                                .set_resource_records(Some(
                                                    records.into_iter().map(|r| {
                                                        ResourceRecord::builder()
                                                            .value(r.value)
                                                            .build()
                                                    }).collect()
                                                ))
                                                .build()
                                        )
                                        .build();

                                    match client
                                        .change_resource_record_sets()
                                        .hosted_zone_id(&hosted_zone_id)
                                        .change_batch(
                                            aws_sdk_route53::types::ChangeBatch::builder()
                                                .changes(change)
                                                .build()
                                        )
                                        .send()
                                        .await
                                    {
                                        Ok(_) => tracing::info!("[SRV] Deleted SRV record: {}", record_name),
                                        Err(e) => tracing::warn!("[SRV] Failed to delete SRV record: {}", e),
                                    }
                                }
                            }
                        }
                    }
                }
                Err(e) => tracing::warn!("[SRV] Failed to list records for deletion: {}", e),
            }
        }

        #[cfg(not(feature = "route53"))]
        {
            tracing::warn!(
                "[SRV] aws-sdk-route53 not available (feature 'route53' not enabled). \
                 Would delete: {}",
                record_name
            );
        }
    }
    ```
