    import { GuildMember, TextChannel, EmbedBuilder } from 'discord.js';
    import { getWelcomeSettings, getInterestList } from '../api/guild-settings';
    import { upsertUser } from '../api/users';

    export async function sendWelcome(member: GuildMember): Promise<void> {
      const guildId = member.guild.id;

      // 1. Fetch guild welcome settings from API (via Plan 02's settings API)
      const settings = await getWelcomeSettings(guildId);
      if (!settings.channelId) return; // No welcome channel configured — skip

      const channel = member.guild.channels.cache.get(settings.channelId.toString()) as TextChannel;
      if (!channel) return; // Channel doesn't exist — skip gracefully

      const lang = settings.language || 'en';
      const greeting = settings.greetingMessage || '';

      // 2. Multi-message sequence per D-01
      // Message 1: Text greeting with mention
      await channel.send(getLocalizedGreeting(lang, member));

      // Message 2: Embed with links, rules, server info per D-03
      const infoEmbed = buildInfoEmbed(lang, member, greeting);
      await channel.send({ embeds: [infoEmbed] });

      // Message 3: Interest selection trigger per D-07
      const interestButton = buildInterestTrigger(lang);
      await channel.send({ components: [interestButton] });

      // 3. Assign default role per D-17
      if (settings.defaultRoleId) {
        const role = member.guild.roles.cache.get(settings.defaultRoleId.toString());
        if (role && !member.roles.cache.has(role.id)) {
          await member.roles.add(role, 'Default role on join');
          // Log per D-20
          await logRoleAssignment(member.guild, settings.logChannelId, member, role, 'default');
        }
      }

      // 4. Track user in DB
      await upsertUser(member.id, guildId, member.user.username, member.user.globalName);
    }

    **Part A: Role manager service (`bot/src/services/role-manager.ts`)**

    ```typescript
    import { Guild, GuildMember, Role, TextChannel, EmbedBuilder } from 'discord.js';
    import { getWelcomeSettings, getInterestList } from '../api/guild-settings';
    import { updateUserInterests } from '../api/users';

    // Assign default role on join (called from welcome service)
    export async function assignDefaultRole(
      member: GuildMember, roleId: string
    ): Promise<boolean> { ... }

    // Assign interest roles after user makes selection
    // Per D-19: append-only — never removes manually-assigned roles
    export async function assignInterestRoles(
      member: GuildMember,
      selectedKeys: string[],
      guildId: string
    ): Promise<{ assigned: string[]; failed: string[] }> {
      const interests = await getInterestList(guildId);
      const assigned: string[] = [];
      const failed: string[] = [];

      for (const key of selectedKeys) {
        const interest = interests.find(i => i.interestKey === key);
        if (!interest || !interest.roleId) {
          failed.push(key);
          continue;
        }

        const role = member.guild.roles.cache.get(interest.roleId.toString());
        if (!role) {
          failed.push(key);
          continue;
        }

        // D-19: Skip if already has the role (no-op)
        if (member.roles.cache.has(role.id)) {
          assigned.push(key);
          continue;
        }

        try {
          await member.roles.add(role, `Interest role: ${interest.displayName}`);
          assigned.push(key);

          // D-20: Log assignment
          const settings = await getWelcomeSettings(guildId);
          if (settings.logChannelId) {
            await logRoleAssignment(member.guild, settings.logChannelId, member, role, 'interest');
          }
        } catch {
          failed.push(key);
        }
      }

      return { assigned, failed };
    }

    // Log role assignment to log channel per D-20
    async function logRoleAssignment(
      guild: Guild,
      logChannelId: string | bigint,
      member: GuildMember,
      role: Role,
      type: 'default' | 'interest'
    ): Promise<void> {
      const channel = guild.channels.cache.get(logChannelId.toString()) as TextChannel;
      if (!channel) return;

      const embed = new EmbedBuilder()
        .setTitle('🏷️ Role Assigned')
        .setDescription(`**User:** ${member.user.tag} (${member.id})\n**Role:** ${role.name} (${role.id})\n**Type:** ${type}\n**Time:** <t:${Math.floor(Date.now() / 1000)}:F>`)
        .setColor(0x57F287)
        .setTimestamp();

      await channel.send({ embeds: [embed] }).catch(() => {});
    }
    ```

    ```typescript
    import { StringSelectMenuBuilder, StringSelectMenuInteraction, ActionRowBuilder } from 'discord.js';
    import { getInterestList } from '../../api/guild-settings';
    import { assignInterestRoles } from '../../services/role-manager';
    import { updateUserInterests } from '../../api/users';

    export const INTEREST_PICK_ID = 'interest:pick';

    export async function handleInterestSelect(interaction: StringSelectMenuInteraction): Promise<void> {
      await interaction.deferReply({ ephemeral: true });

      const selectedKeys = interaction.values; // Array of selected interest keys
      const guildId = interaction.guildId!;
      const member = interaction.member!;

      // Assign roles per D-19 (append-only)
      const { assigned, failed } = await assignInterestRoles(member, selectedKeys, guildId);

      // Update user's interest selections in DB
      const interests = await getInterestList(guildId);
      const assignedNames = assigned
        .map(key => interests.find(i => i.interestKey === key)?.displayName)
        .filter(Boolean);
      const failedNames = failed
        .map(key => interests.find(i => i.interestKey === key)?.displayName)
        .filter(Boolean);

      // Update user record
      await updateUserInterests(interaction.user.id, guildId, selectedKeys);

      // Build response message
      let response = '';
      if (assigned.length > 0) {
        response += `✅ **Roles assigned:** ${assignedNames.join(', ')}\n`;
      }
      if (failed.length > 0) {
        response += `❌ **Could not assign:** ${failedNames.join(', ')}\n`;
      }
      response += '\nYou can change your interests anytime by clicking the button again.';

      await interaction.editReply({ content: response });
    }
    ```

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-03-01 | Spoofing | guildMemberAdd event | accept | Discord Gateway events are authenticated by the Gateway connection. No additional verification needed. |
| T-03-02 | Tampering | Select menu values | mitigate | Select menu values are constrained to configured interest keys (Discord validates against option values). Bot ignores unknown values. |
| T-03-03 | Repudiation | Role assignments | mitigate | All role assignments logged to log channel with user, role, type, and timestamp per D-20. |
| T-03-04 | Information Disclosure | Welcome channel visibility | accept | Welcome message is public by design (sent to a server channel). No sensitive info in welcome content. |
| T-03-05 | Denial of Service | Bulk role assignment | mitigate | Role assignments are made sequentially with error handling per role. Discord.js REST handles rate limits. Max 5 interests = max 5 API calls. |
| T-03-06 | Elevation of Privilege | Role assignment via select menu | mitigate | Bot only assigns roles mapped to interests in DB. Interest→role mapping is admin-configured via Plan 02 settings. Users cannot request arbitrary roles. |
| T-03-07 | Tampering | customId impersonation | mitigate | Select menu customId includes userId suffix (`interest:pick:{userId}`). Handler verifies interaction.user.id matches the embedded userId. |
