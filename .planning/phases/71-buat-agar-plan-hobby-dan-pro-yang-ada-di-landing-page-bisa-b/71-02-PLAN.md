**B. Create `app/src/pages/dashboard/WelcomeModal.jsx`** (per D-08: checkout callback with success/cancel detection; D-12: welcome modal shows plan name, features unlocked, limits, and "Start creating servers" CTA).

Follow the dashboard's existing modal pattern from `InviteFriendsModal.jsx`. The component uses dark theme classes (`bg-gray-800`, `text-white`, `bg-gray-900`) matching the dashboard design system.

```jsx
import { useState, useEffect } from 'react'
import { useSearchParams } from 'react-router-dom'
import { useUIStore } from '../../store/uiStore'
import { billingApi } from '../../lib/api'
import { CheckCircle2, Sparkles, X } from 'lucide-react'

export default function WelcomeModal() {
  const [searchParams] = useSearchParams()
  const [isOpen, setIsOpen] = useState(false)
  const [subscription, setSubscription] = useState(null)
  const [isLoading, setIsLoading] = useState(true)
  const { addToast } = useUIStore()

  // Check for checkout status on mount
  useEffect(() => {
    const checkoutStatus = searchParams.get('checkout')
    
    if (checkoutStatus === 'success') {
      // Fetch subscription to get plan name and details
      const loadSubscription = async () => {
        try {
          const response = await billingApi.getCurrentSubscription()
          const sub = response?.data || response || null
          setSubscription(sub)
          setIsOpen(true)
        } catch (err) {
          console.error('Failed to load subscription for welcome modal:', err)
          setIsOpen(true) // Still show modal even if fetch fails
        } finally {
          setIsLoading(false)
        }
      }
      loadSubscription()
      
      // Clean URL to prevent re-trigger on refresh (see Pitfall 3)
      window.history.replaceState({}, '', '/dashboard')
    } else if (checkoutStatus === 'cancelled' || checkoutStatus === 'canceled') {
      // Show cancellation toast
      addToast({ type: 'error', message: 'Checkout canceled. You can try again anytime.' })
      window.history.replaceState({}, '', '/dashboard')
    }
  }, [searchParams, addToast])

  const handleClose = () => {
    setIsOpen(false)
  }

  if (!isOpen) return null

  const planName = subscription?.plan?.display_name || subscription?.plan?.name || 'Premium'
  const planFeatures = subscription?.plan?.features || []
  const planLimits = subscription?.plan?.limits || {}

  return (
    <div
      className="fixed inset-0 bg-black/60 flex items-center justify-center z-50 p-4"
      onClick={handleClose}
      role="dialog"
      aria-modal="true"
    >
      <div
        className="bg-gray-800 border border-gray-700 rounded-lg p-8 max-w-md w-full space-y-6"
        onClick={(e) => e.stopPropagation()}
      >
        {/* Close button */}
        <div className="flex items-center justify-between">
          <div className="w-12 h-12 bg-green-600/20 rounded-full flex items-center justify-center">
            <Sparkles className="w-6 h-6 text-green-400" />
          </div>
          <button
            onClick={handleClose}
            className="text-gray-400 hover:text-white text-xl leading-none"
            aria-label="Close"
          >
            <X className="w-5 h-5" />
          </button>
        </div>

        {/* Content */}
        <div className="space-y-2">
          <h2 className="text-2xl font-bold text-white">
            Welcome to {planName}!
          </h2>
          <p className="text-gray-400">
            You're all set. Here's what you unlocked:
          </p>
        </div>

        {/* Features */}
        {planFeatures.length > 0 && (
          <ul className="space-y-2">
            {planFeatures.slice(0, 6).map((feature, i) => (
              <li key={i} className="flex items-start gap-3 text-sm text-gray-300">
                <CheckCircle2 className="w-4 h-4 text-green-400 shrink-0 mt-0.5" />
                <span>{feature}</span>
              </li>
            ))}
            {planFeatures.length > 6 && (
              <li className="text-sm text-gray-500 pl-7">
                +{planFeatures.length - 6} more features
              </li>
            )}
          </ul>
        )}

        {/* Limits */}
        {Object.keys(planLimits).length > 0 && (
          <div className="bg-gray-900 rounded-lg p-4 space-y-2">
            <p className="text-sm font-semibold text-gray-400 uppercase tracking-wider text-xs">Your Limits</p>
            <div className="grid grid-cols-2 gap-2">
              {Object.entries(planLimits).slice(0, 4).map(([key, value]) => (
                <div key={key}>
                  <span className="text-xs text-gray-500 block">
                    {key.replace(/_/g, ' ').replace(/\b\w/g, c => c.toUpperCase())}
                  </span>
                  <span className="text-sm font-semibold text-white">
                    {value === -1 ? 'Unlimited' : value}
                  </span>
                </div>
              ))}
            </div>
          </div>
        )}

        {/* CTA */}
        <button
          onClick={handleClose}
          className="w-full py-3 px-4 rounded-xl font-semibold text-white bg-gradient-to-r from-blue-600 to-cyan-500 hover:from-blue-500 hover:to-cyan-400 transition-all duration-200 shadow-lg shadow-blue-600/25"
        >
          Start creating servers
        </button>
      </div>
    </div>
  )
}
```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| URL params (user-facing) | `?checkout=success` and `?checkout=canceled` params are visible in browser URL |
| Dashboard WelcomeModal | Cosmetic UI component, no backend calls triggered by modal display |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-71-06 | Spoofing | `?checkout=success` param | accept | Param is cosmetic only. No backend call triggered by WelcomeModal display. URL is cleaned via `window.history.replaceState()` immediately on detection. No data exposure risk. |
| T-71-07 | Information Disclosure | Subscription data in WelcomeModal | mitigate | WelcomeModal fetches subscription via `billingApi.getCurrentSubscription()` which requires backend `VerifiedUser` middleware. Only the authenticated user's subscription is returned. |
| T-71-08 | CSRF | `POST /api/v1/billing/portal` | mitigate | Backend uses cookie-based auth. The dashboard's `api.js` sends `Authorization: Bearer` token from auth store. Existing ProtectedRoute middleware prevents unauthenticated access. |
</threat_model>
