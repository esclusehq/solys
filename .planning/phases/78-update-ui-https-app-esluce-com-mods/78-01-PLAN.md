    /// Returns available game versions by proxying the Modrinth tag list.
    pub async fn get_game_versions(
        State(_container): State<ApiState>,
    ) -> Result<impl IntoResponse, AppError> {
        let client = reqwest::Client::builder()
            .user_agent("DevNode-V2/2.0.0 (github.com/rhnbztnl/DevNode-V2)")
            .build()
            .map_err(|e| AppError::InternalError(anyhow::anyhow!(e)))?;

        let url = "https://api.modrinth.com/v2/tag/game_version";
        let response = client
            .get(url)
            .send()
            .await
            .map_err(|e| AppError::InternalError(anyhow::anyhow!(e)))?;

        let versions = response
            .json::<serde_json::Value>()
            .await
            .map_err(|e| AppError::InternalError(anyhow::anyhow!(e)))?;

        Ok(Json(ApiResponse::success(versions)))
