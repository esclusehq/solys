    ```jsx
    import { useState, useRef, useEffect } from 'react'
    import { X, ChevronLeft, ChevronRight } from 'lucide-react'
    import { modsApi } from '../../api/templatesApi'
    import ModSearchResult from '../../components/ModSearchResult'
    import { EscluseSpinner } from '../../components/SkeletonLoader'

    const FALLBACK_VERSIONS = ['1.21', '1.20.4', '1.20.1', '1.19.4', '1.18.2', '1.16.5']

    export default function ModBrowserPage() {
      const [results, setResults] = useState([])
      const [total, setTotal] = useState(0)
      const [loading, setLoading] = useState(false)
      const [error, setError] = useState(null)
      const [query, setQuery] = useState('')
      const [version, setVersion] = useState('')
      const [loader, setLoader] = useState('')
      const [category, setCategory] = useState('')
      const [currentPage, setCurrentPage] = useState(1)
      const [gameVersions, setGameVersions] = useState(FALLBACK_VERSIONS)
      const [showVersionModal, setShowVersionModal] = useState(null)
      const [modalVersions, setModalVersions] = useState([])
      const [modalLoading, setModalLoading] = useState(false)
      const [modalError, setModalError] = useState(null)
      const pageSize = 25
      const debounceRef = useRef(null)

      // Fetch available game versions dynamically on mount (D-09)
      useEffect(() => {
        ;(async () => {
          try {
            const result = await modsApi.getGameVersions()
            const versions = result?.versions || result || []
            if (Array.isArray(versions) && versions.length > 0) {
              const extracted = versions.map(v => typeof v === 'string' ? v : v.version).filter(Boolean)
              if (extracted.length > 0) setGameVersions(extracted)
              else if (typeof versions[0] === 'string') setGameVersions(versions)
            }
          } catch {
            // Fallback to hardcoded list — non-blocking
          }
        })()
      }, [])

      useEffect(() => {
        return () => clearTimeout(debounceRef.current)
      }, [])

      const executeSearch = async (q, v, l, c, page) => {
        if (!q || q.trim().length < 2) {
          setResults([])
          setTotal(0)
          return
        }
        try {
          setLoading(true)
          setError(null)
          const offset = (page - 1) * pageSize
          const params = { q, offset: offset.toString(), sort: 'downloads' }
          if (v) params.version = v
          if (l) params.loader = l
          if (c) params.project_type = c  // CRITICAL: use project_type (not category) per backend PluginSearchQuery
          const data = await modsApi.search(params)
          setResults(data.plugins || data.hits || [])
          setTotal(data.total || 0)
        } catch (e) {
          setError(e.message || 'Failed to search mods')
          if (results.length === 0) setResults([])
        } finally {
          setLoading(false)
        }
      }

      const triggerSearch = (q = query, v = version, l = loader, c = category, page = 1) => {
        clearTimeout(debounceRef.current)
        debounceRef.current = setTimeout(() => {
          executeSearch(q, v, l, c, page)
        }, 300)
      }

      const handleSearchChange = (e) => {
        const val = e.target.value
        setQuery(val)
        setCurrentPage(1)
        triggerSearch(val, version, loader, category, 1)
      }

      const handleVersionChange = (e) => {
        setVersion(e.target.value)
        setCurrentPage(1)
        triggerSearch(query, e.target.value, loader, category, 1)
      }

      const handleLoaderChange = (e) => {
        setLoader(e.target.value)
        setCurrentPage(1)
        triggerSearch(query, version, e.target.value, category, 1)
      }

      const handleCategoryChange = (e) => {
        setCategory(e.target.value)
        setCurrentPage(1)
        triggerSearch(query, version, loader, e.target.value, 1)
      }

      const handlePageChange = (page) => {
        if (page < 1 || (totalPages > 0 && page > totalPages)) return
        setCurrentPage(page)
        executeSearch(query, version, loader, category, page)
      }

      const handleViewVersions = async (mod) => {
        setShowVersionModal(mod)
        setModalVersions([])
        setModalError(null)
        setModalLoading(true)
        try {
          const versions = await modsApi.getVersions(mod.project_id)
          setModalVersions(Array.isArray(versions) ? versions : [])
        } catch (e) {
          setModalError(e.message || 'Failed to load versions')
        } finally {
          setModalLoading(false)
        }
      }

      const closeVersionModal = () => {
        setShowVersionModal(null)
        setModalVersions([])
        setModalError(null)
      }

      const handleInstallFromVersion = (mod, versionData) => {
        closeVersionModal()
        // Opens install modal with pre-selected version — wired in Plan 03
      }

      const totalPages = Math.max(1, Math.ceil(total / pageSize))

      const getPageNumbers = () => {
        const pages = []
        const maxVisible = 7
        if (totalPages <= maxVisible) {
          for (let i = 1; i <= totalPages; i++) pages.push(i)
        } else {
          pages.push(1)
          let start = Math.max(2, currentPage - 2)
          let end = Math.min(totalPages - 1, currentPage + 2)
          if (currentPage <= 3) { start = 2; end = Math.min(6, totalPages - 1) }
          if (currentPage >= totalPages - 2) { start = Math.max(2, totalPages - 5); end = totalPages - 1 }
          if (start > 2) pages.push('...')
          for (let i = start; i <= end; i++) pages.push(i)
          if (end < totalPages - 1) pages.push('...')
          pages.push(totalPages)
        }
        return pages
      }

      return (
        <div className="p-6">
          {/* Header */}
          <h2 className="text-2xl font-bold text-[var(--color-text-main)] mb-2">Mod Browser</h2>
          <p className="text-xs text-[var(--color-text-muted)] mb-5">
            Browse and search mods from Modrinth. Click <strong>Versions</strong> to see available versions, or <strong>Add</strong> to include in a collection.
          </p>

          {/* Filters */}
          <div className="flex flex-wrap gap-3 mb-6">
            <input type="text" value={query} onChange={handleSearchChange}
                   placeholder="Search mods (min 2 characters)..."
                   className="flex-1 min-w-[200px] px-4 py-2.5 rounded-lg text-sm bg-[var(--color-cosmic-card)]/60
                              border border-[var(--color-cosmic-border)] text-[var(--color-text-main)]
                              placeholder:text-[var(--color-text-muted)]" />

            <select value={category} onChange={handleCategoryChange}
                    className="px-4 py-2.5 rounded-lg text-sm bg-[var(--color-cosmic-card)]/60
                               border border-[var(--color-cosmic-border)] text-[var(--color-text-main)]">
              <option value="">All Categories</option>
              <option value="bukkit">Bukkit</option>
              <option value="mod">Mod</option>
              <option value="datapack">Datapack</option>
              <option value="resourcepack">Resourcepack</option>
              <option value="shader">Shader</option>
            </select>

            <select value={version} onChange={handleVersionChange}
                    className="px-4 py-2.5 rounded-lg text-sm bg-[var(--color-cosmic-card)]/60
                               border border-[var(--color-cosmic-border)] text-[var(--color-text-main)]">
              <option value="">All Versions</option>
              {gameVersions.map(gv => (
                <option key={gv} value={gv}>{gv}</option>
              ))}
            </select>

            <select value={loader} onChange={handleLoaderChange}
                    className="px-4 py-2.5 rounded-lg text-sm bg-[var(--color-cosmic-card)]/60
                               border border-[var(--color-cosmic-border)] text-[var(--color-text-main)]">
              <option value="">All Loaders</option>
              <option value="forge">Forge</option>
              <option value="fabric">Fabric</option>
              <option value="quilt">Quilt</option>
              <option value="neoforge">NeoForge</option>
              <option value="paper">Paper</option>
              <option value="spigot">Spigot</option>
              <option value="purpur">Purpur</option>
            </select>
          </div>

          {/* Error */}
          {error && (
            <div className="mb-4 px-4 py-3 rounded-lg text-sm font-medium bg-red-500/10 text-[var(--color-cosmic-red)] border border-red-500/20 flex items-center justify-between">
              <span>Failed to search mods. Please try again.</span>
              <button onClick={() => { setError(null); executeSearch(query, version, loader, category, currentPage) }}
                      className="ml-4 px-3 py-1 rounded bg-red-500/20 hover:bg-red-500/30 text-[var(--color-cosmic-red)] text-xs font-semibold">
                Retry
              </button>
            </div>
          )}

          {/* Loading */}
          {loading && <EscluseSpinner />}

          {/* Results */}
          {!loading && results.length > 0 && (
            <>
              <p className="text-xs text-[var(--color-text-muted)] mb-4">
                <strong className="text-[var(--color-text-main)]">{total}</strong> results found
              </p>
              <div className="grid grid-cols-1 md:grid-cols-2 gap-4">
                {results.map((mod, i) => (
                  <ModSearchResult key={mod.project_id || i} mod={mod}
                                   onViewVersions={handleViewVersions}
                                   onAdd={() => {}} />  {/* Wired in Plan 03 */}
                ))}
              </div>

              {/* Pagination */}
              {totalPages > 1 && (
                <div className="flex items-center justify-center gap-1.5 mt-6">
                  <button onClick={() => handlePageChange(currentPage - 1)}
                          disabled={currentPage <= 1}
                          className="p-2 rounded text-sm text-[var(--color-text-muted)] hover:text-white disabled:opacity-40 disabled:cursor-not-allowed">
                    <ChevronLeft size={16} />
                  </button>
                  {getPageNumbers().map((p, i) =>
                    p === '...' ? (
                      <span key={`ellipsis-${i}`} className="px-2 text-[var(--color-text-muted)] text-sm">...</span>
                    ) : (
                      <button key={p} onClick={() => handlePageChange(p)}
                              className={`w-8 h-8 rounded text-sm transition-colors ${
                                p === currentPage
                                  ? 'bg-[var(--color-cosmic-cyan)]/10 text-[var(--color-cosmic-cyan)] font-bold'
                                  : 'text-[var(--color-text-muted)] hover:text-white'
                              }`}>
                        {p}
                      </button>
                    )
                  )}
                  <button onClick={() => handlePageChange(currentPage + 1)}
                          disabled={currentPage >= totalPages}
                          className="p-2 rounded text-sm text-[var(--color-text-muted)] hover:text-white disabled:opacity-40 disabled:cursor-not-allowed">
                    <ChevronRight size={16} />
                  </button>
                </div>
              )}
            </>
          )}

          {/* Empty state (no results for query) */}
          {!loading && !error && results.length === 0 && query.length >= 2 && (
            <div className="text-center py-12">
              <p className="text-[var(--color-text-muted)]">No mods found matching your search.</p>
            </div>
          )}

          {/* Initial state */}
          {!loading && !error && query.length < 2 && (
            <div className="text-center py-12">
              <p className="text-[var(--color-text-muted)]">Enter at least 2 characters to search mods.</p>
            </div>
          )}

          {/* Version List Modal (D-06, D-07) */}
          {showVersionModal && (
            <div className="fixed inset-0 bg-black/60 flex items-center justify-center z-50 p-4"
                 onClick={closeVersionModal}
                 role="dialog" aria-modal="true">
              <div className="bg-[var(--color-nebula)] border border-[var(--color-cosmic-border)] rounded-xl p-6 max-w-lg w-full space-y-4 shadow-xl max-h-[80vh] flex flex-col"
                   onClick={(e) => e.stopPropagation()}>
                {/* Header */}
                <div className="flex items-center justify-between flex-shrink-0">
                  <h3 className="text-sm font-bold text-[var(--color-text-main)]">
                    Versions for <span className="text-[var(--color-cosmic-cyan)]">{showVersionModal.title}</span>
                  </h3>
                  <button onClick={closeVersionModal}
                          className="p-1 rounded hover:bg-[rgba(255,255,255,0.05)] text-[var(--color-text-muted)] hover:text-white">
                    <X size={18} />
                  </button>
                </div>

                {/* Content */}
                <div className="overflow-y-auto flex-1 space-y-2">
                  {modalLoading ? (
                    <div className="flex justify-center py-8"><EscluseSpinner /></div>
                  ) : modalError ? (
                    <div className="text-center py-8">
                      <p className="text-[var(--color-cosmic-red)] text-sm">{modalError}</p>
                      <button onClick={() => handleViewVersions(showVersionModal)}
                              className="mt-3 text-xs text-[var(--color-cosmic-cyan)] hover:underline">
                        Retry
                      </button>
                    </div>
                  ) : modalVersions.length === 0 ? (
                    <p className="text-center py-8 text-[var(--color-text-muted)] text-sm">No versions available for this mod.</p>
                  ) : (
                    modalVersions.map((ver) => (
                      <div key={ver.id}
                           className="flex items-center justify-between p-3 rounded-lg bg-[rgba(255,255,255,0.02)] border border-[var(--color-cosmic-border)]">
                        <div className="min-w-0 flex-1">
                          <div className="flex items-center gap-2">
                            <span className="font-mono text-sm font-bold text-[var(--color-text-main)]">{ver.version_number}</span>
                            <span className="text-xs text-[var(--color-text-muted)]">{ver.name}</span>
                          </div>
                          <div className="flex items-center gap-2 mt-1">
                            <span className="text-[10px] text-[var(--color-text-muted)]">
                              {ver.date_published ? new Date(ver.date_published).toLocaleDateString() : '—'}
                            </span>
                            {(ver.loaders || []).map(ld => (
                              <span key={ld}
                                    className="px-1.5 py-0.5 text-[10px] rounded bg-[rgba(255,255,255,0.05)] text-[var(--color-text-muted)]">
                                {ld}
                              </span>
                            ))}
                          </div>
                        </div>
                        <button onClick={() => handleInstallFromVersion(showVersionModal, ver)}
                                className="ml-3 px-3 py-1.5 text-xs rounded font-semibold whitespace-nowrap
                                           bg-[var(--color-cosmic-cyan)] text-[#07090e]
                                           hover:bg-[#22dcf0] shadow-[0_0_15px_rgba(13,223,242,0.3)]">
                          Install
                        </button>
                      </div>
                    ))
                  )}
                </div>
              </div>
            </div>
          )}
        </div>
      )
    }
    ```
