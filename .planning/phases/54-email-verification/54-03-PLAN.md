  ```jsx
  import { useState, useEffect } from 'react'
  import { Link } from 'react-router-dom'
  import { useAuthStore } from '../store/authStore'
  import { useUIStore } from '../store/uiStore'

  export default function EmailVerificationBanner() {
      const { user, resendVerification } = useAuthStore()
      const { addToast } = useUIStore()
      const [cooldown, setCooldown] = useState(0)
      const [maxedOut, setMaxedOut] = useState(false)

      // D-07: Countdown timer
      useEffect(() => {
          if (cooldown <= 0) return
          const timer = setInterval(() => {
              setCooldown(prev => prev - 1)
          }, 1000)
          return () => clearInterval(timer)
      }, [cooldown])

      // D-02: Only show if authenticated AND not email verified
      if (!user || user.email_verified) return null

      const handleResend = async () => {
          // D-07: Clicking during cooldown shows tooltip with remaining time
          // Button stays enabled — clicking triggers feedback instead of silent ignore
          if (cooldown > 0) {
              addToast({ type: 'info', message: `Please wait ${cooldown}s before requesting another verification email.` })
              return
          }
          try {
              const result = await resendVerification()
              setCooldown(60) // D-05: 60s cooldown
              // Toast is handled by authStore
          } catch (err) {
              // Check if maxed out
              if (err.message && err.message.includes('Contact support')) {
                  setMaxedOut(true)
              }
              // Check for cooldown error with remaining time
              if (err.message && err.message.includes('wait')) {
                  setCooldown(60) // start countdown anyway
              }
              // Toast is handled by authStore
          }
      }

      return (
          <div className="bg-yellow-500/10 border-b border-yellow-500/20 px-4 py-2">
              <div className="flex items-center justify-between max-w-7xl mx-auto">
                  <p className="text-yellow-400 text-sm">
                      ⚠ Verify your email — Check your inbox at <strong>{user.email}</strong>
                  </p>
                  <div className="flex items-center gap-3">
                      {maxedOut ? (
                          <span className="text-gray-400 text-sm">Contact support</span>
                      ) : (
                          <>
                              <button
                                  onClick={handleResend}
                                  // D-07: Button stays enabled during cooldown.
                                  // Clicking shows tooltip (toast) with remaining time instead of silent disabled state.
                                  className="text-yellow-400 hover:text-yellow-300 text-sm underline"
                                  title={cooldown > 0 ? `Resend available in ${cooldown}s` : 'Send verification email again'}
                              >
                                  {cooldown > 0 ? `Resend in ${cooldown}s` : 'Resend Email'}
                              </button>
                              <Link
                                  to="/settings"
                                  className="text-gray-400 hover:text-white text-sm"
                              >
                                  Change Email
                              </Link>
                          </>
                      )}
                  </div>
              </div>
          </div>
      )
  }
  ```

  - **D-06:** After 5 attempts, shows "Contact support" (maxedOut state)
  - **D-07:** Button stays enabled during cooldown (no `disabled` prop). Clicking during cooldown shows toast with remaining time instead of silently ignoring the click. Countdown timer visible in button text. Title tooltip shows remaining seconds.

  ```jsx
  import { useState, useEffect } from 'react'
  import { useAuthStore } from '../store/authStore'
  import { useUIStore } from '../store/uiStore'

  export default function EmailVerificationDialog({ onClose }) {
      const { user, resendVerification } = useAuthStore()
      const { addToast } = useUIStore()
      const [cooldown, setCooldown] = useState(0)
      const [maxedOut, setMaxedOut] = useState(false)

      // D-07: Countdown timer
      useEffect(() => {
          if (cooldown <= 0) return
          const timer = setInterval(() => {
              setCooldown(prev => prev - 1)
          }, 1000)
          return () => clearInterval(timer)
      }, [cooldown])

      const handleResend = async () => {
          // D-07: Clicking during cooldown shows tooltip with remaining time
          // Button stays enabled — clicking triggers toast feedback instead of silent ignore
          if (cooldown > 0) {
              addToast({ type: 'info', message: `Please wait ${cooldown}s before requesting another verification email.` })
              return
          }
          try {
              await resendVerification()
              setCooldown(60)
              addToast({ type: 'success', message: 'Verification email sent!' })
          } catch (err) {
              if (err.message && err.message.includes('Contact support')) {
                  setMaxedOut(true)
              }
              if (err.message && err.message.includes('wait')) {
                  setCooldown(60)
              }
          }
      }

      return (
          <div className="fixed inset-0 bg-black/50 z-50 flex items-center justify-center">
              <div className="bg-gray-800 rounded-lg p-6 max-w-md w-full mx-4">
                  <h2 className="text-lg font-semibold text-white mb-2">
                      Email Verification Required
                  </h2>
                  <p className="text-gray-400 mb-2">
                      Verify your email before using this feature.
                  </p>
                  <p className="text-gray-500 text-sm mb-6">
                      Check your inbox at <strong className="text-gray-300">{user?.email}</strong>
                  </p>
                  
                  <div className="flex justify-end gap-3">
                      <button
                          onClick={onClose}
                          className="px-4 py-2 text-gray-400 hover:text-white text-sm"
                      >
                          Cancel
                      </button>
                      {maxedOut ? (
                          <span className="px-4 py-2 text-gray-500 text-sm">
                              Contact support
                          </span>
                      ) : (
                          <button
                              onClick={handleResend}
                              // D-07: Button stays enabled during cooldown.
                              // Clicking shows toast with remaining time instead of silent disabled state.
                              className="px-4 py-2 bg-yellow-600 text-white rounded hover:bg-yellow-700 text-sm"
                          >
                              {cooldown > 0 ? `Resend in ${cooldown}s` : 'Resend Email'}
                          </button>
                      )}
                  </div>
              </div>
          </div>
      )
  }
  ```

  Key behaviors per decisions:
  - **D-03:** Modal overlay with "Email verification required" message
  - **D-07:** Inline resend with cooldown timer (same pattern as banner — button stays enabled, clicking during cooldown shows toast with remaining time)
  - **D-06:** After max attempts, shows "Contact support" instead of resend button
  - The dialog accepts an `onClose` prop to let parent control visibility
