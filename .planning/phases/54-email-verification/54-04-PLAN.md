  ```jsx
  import { useAuthStore } from '../store/authStore'
  import EmailVerificationDialog from './EmailVerificationDialog'

  export default function VerifiedRoute({ children }) {
      const { user } = useAuthStore()
      const [showDialog, setShowDialog] = useState(false)

      // D-10: Frontend gating — defense in depth
      // Backend VerifiedUser extractor is the real enforcement
      // D-08: Gated features (wrapped by this component)
      // D-09: Ungated features should NOT use this component

      if (!user) return null // ProtectedRoute handles unauthenticated

      if (user.email_verified) {
          return children
      }

      // D-03: Show blocking dialog for unverified users
      // The feature page is still mounted (for structured layout) but the dialog blocks interaction
      return (
          <div className="relative">
              {children}
              {/* Semi-transparent overlay */}
              <div className="fixed inset-0 z-40 bg-gray-900/50" />
              <EmailVerificationDialog onClose={() => setShowDialog(false)} />
          </div>
      )
  }
  ```
