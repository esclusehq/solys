```rust
pub async fn get_node_limit(&self, user_id: Uuid) -> Result<i32> {
    let subscription = self.subscription_repo.find_by_user_id(user_id).await?;
    
    if let Some(sub) = subscription {
        if sub.status != "active" {
            return Ok(1); // Free tier for inactive
        }
        let plan = self.plan_repo.find_by_id(sub.plan_id).await?;
        if let Some(plan) = plan {
            let limits: serde_json::Value = plan.limits;
            if let Some(max_nodes) = limits.get("max_nodes").and_then(|v| v.as_i64()) {
                return Ok(max_nodes as i32);
            }
        }
    }
    
    Ok(1) // Default to 1 for free tier
}
```
