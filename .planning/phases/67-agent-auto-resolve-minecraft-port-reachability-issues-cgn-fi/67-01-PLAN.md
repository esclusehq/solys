```rust
use anyhow::{Context, Result};
use chrono::{DateTime, Utc};
use sqlx::{PgPool, Row};
use uuid::Uuid;
use crate::domain::entities::connectivity_audit_log::ConnectivityAuditLog;

pub struct PostgresConnectivityAuditLogRepository {
    pool: PgPool,
}

impl PostgresConnectivityAuditLogRepository {
    pub fn new(pool: PgPool) -> Self { Self { pool } }

    pub async fn insert(&self, log: &ConnectivityAuditLog) -> Result<()> {
        sqlx::query(
            r#"INSERT INTO connectivity_audit_log
                 (id, server_id, node_id, event_type, command, status, details, created_at)
               VALUES ($1, $2, $3, $4, $5, $6, $7, $8)"#,
        )
        .bind(log.id)
        .bind(log.server_id)
        .bind(log.node_id)
        .bind(&log.event_type)
        .bind(&log.command)
        .bind(&log.status)
        .bind(&log.details)
        .bind(log.created_at)
        .execute(&self.pool)
        .await
        .context("Failed to insert connectivity audit log")?;
        Ok(())
    }

    pub async fn list_by_server(
        &self, server_id: Uuid, limit: i64, offset: i64,
    ) -> Result<Vec<ConnectivityAuditLog>> {
        let rows = sqlx::query(
            r#"SELECT id, server_id, node_id, event_type, command, status, details, created_at
               FROM connectivity_audit_log
               WHERE server_id = $1
               ORDER BY created_at DESC
               LIMIT $2 OFFSET $3"#,
        )
        .bind(server_id)
        .bind(limit)
        .bind(offset)
        .fetch_all(&self.pool)
        .await
        .context("Failed to list connectivity audit logs")?;
        let mut out = Vec::with_capacity(rows.len());
        for r in rows {
            out.push(ConnectivityAuditLog {
                id: r.try_get("id")?,
                server_id: r.try_get("server_id")?,
                node_id: r.try_get("node_id")?,
                event_type: r.try_get("event_type")?,
                command: r.try_get("command")?,
                status: r.try_get("status")?,
                details: r.try_get("details")?,
                created_at: r.try_get("created_at")?,
            });
        }
        Ok(out)
    }

    pub async fn count_by_server(&self, server_id: Uuid) -> Result<i64> {
        let count: i64 = sqlx::query_scalar(
            "SELECT COUNT(*) FROM connectivity_audit_log WHERE server_id = $1",
        )
        .bind(server_id)
        .fetch_one(&self.pool)
        .await
        .context("Failed to count connectivity audit logs")?;
        Ok(count)
    }
}
```

NOTE: This impl does NOT use `#[async_trait]` because the project does not implement the trait for the concrete repo (the project casts `Arc<PostgresServerRepository>` to `Arc<dyn ServerRepository>` only when the trait is in `domain/`). Since the future service in Plan 03 only needs the concrete `PostgresConnectivityAuditLogRepository` methods, and to match `crash_log_repository.rs` exactly (which is also not async-trait-impl'd), this is correct. If the future service needs the trait, Plan 03 adds `impl ConnectivityAuditLogRepository for PostgresConnectivityAuditLogRepository { ... }` in the same file.
