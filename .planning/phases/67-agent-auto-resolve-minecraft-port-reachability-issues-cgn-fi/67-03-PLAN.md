```rust
//! Phase 67: Connectivity probe + classification + auto-fix dispatcher.
//!
//! Owns the outbound TCP+SLP probe (Java Edition), RakNet ping (Bedrock), the
//! 4-MVP failure-mode classifier, and the safe-to-fix dispatch loop that
//! sends `NodeMessage::ConnectivityFixRequest` to the agent.
//!
//! Also runs a 5-min periodic re-probe of all running servers (spawned in
//! `bootstrap/mod.rs` next to `monitoring_service.start()`).

use std::sync::Arc;
use std::time::Duration;
use anyhow::{anyhow, Result};
use chrono::{DateTime, Utc};
use serde::{Deserialize, Serialize};
use serde_json::{json, Value};
use sqlx::PgPool;
use tokio::io::{AsyncReadExt, AsyncWriteExt};
use tokio::net::{TcpStream, UdpSocket};
use uuid::Uuid;

use crate::application::services::monitoring_service::CrashReportData; // unused — placeholder
use crate::domain::repositories::server_repository::ServerRepository;
use crate::domain::repositories::node_repository::NodeRepository;
use crate::domain::repositories::connectivity_audit_log_repository::ConnectivityAuditLogRepository;
use crate::infrastructure::repositories::sqlx_connectivity_audit_log_repository::PostgresConnectivityAuditLogRepository;
use crate::domain::entities::connectivity_audit_log::ConnectivityAuditLog;
use crate::presentation::ws::node_connection_manager::NodeConnectionManager;
use crate::presentation::ws::node_protocol::NodeMessage;

const PROBE_TIMEOUT: Duration = Duration::from_secs(10);
const PERIODIC_INTERVAL: Duration = Duration::from_secs(300); // 5 min

/// 4-MVP failure modes (D-06). `is_cgnat_detected` is NOT auto-fixable.
#[derive(Debug, Clone, Serialize, Deserialize)]
pub enum FailureMode {
    PortNotBound,
    HostFirewallBlocked,
    CgnatDetected,
    UpnpUnavailable,
}

impl FailureMode {
    pub fn as_str(&self) -> &'static str {
        match self {
            Self::PortNotBound => "PORT_NOT_BOUND",
            Self::HostFirewallBlocked => "HOST_FIREWALL_BLOCKED",
            Self::CgnatDetected => "CGNAT_DETECTED",
            Self::UpnpUnavailable => "UPNP_UNAVAILABLE",
        }
    }
    pub fn is_auto_fixable(&self) -> bool {
        !matches!(self, Self::CgnatDetected)
    }
}

/// Result of a single probe attempt.
#[derive(Debug, Clone, Serialize)]
pub struct ProbeResult {
    pub tcp_ok: bool,
    pub handshake_ok: bool,
    pub status_ok: bool,
    pub version: Option<String>,
    pub players_online: Option<i32>,
    pub players_max: Option<i32>,
    pub latency_ms: Option<u64>,
}

/// Full connectivity state for a server (mirrors servers.connectivity_state JSONB).
#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct ConnectivityState {
    pub status: String,            // "unknown" | "reachable" | "unreachable"
    pub mode: String,               // "direct" | "relay" | "offline"
    pub last_probe_at: Option<DateTime<Utc>>,
    pub failure_mode: Option<String>,
    pub details: Value,             // raw agent diagnostics
    pub last_error: Option<String>,
}

impl Default for ConnectivityState {
    fn default() -> Self {
        Self {
            status: "unknown".into(),
            mode: "direct".into(),
            last_probe_at: None,
            failure_mode: None,
            details: json!({}),
            last_error: None,
        }
    }
}

pub struct ConnectivityService {
    server_repository: Arc<dyn ServerRepository>,
    node_repository: Arc<dyn NodeRepository>,
    node_connection_manager: Arc<NodeConnectionManager>,
    audit_log_repo: Arc<PostgresConnectivityAuditLogRepository>,
    pool: PgPool,
}

impl ConnectivityService {
    pub fn new(
        server_repository: Arc<dyn ServerRepository>,
        node_repository: Arc<dyn NodeRepository>,
        node_connection_manager: Arc<NodeConnectionManager>,
        audit_log_repo: Arc<PostgresConnectivityAuditLogRepository>,
        pool: PgPool,
    ) -> Self {
        Self { server_repository, node_repository, node_connection_manager, audit_log_repo, pool }
    }

    /// Background loop — re-probes all running servers every 5 min.
    pub async fn start(self: Arc<Self>) {
        tracing::info!("[CONNECTIVITY] Starting periodic re-probe loop ({}s)", PERIODIC_INTERVAL.as_secs());
        let mut ticker = tokio::time::interval(PERIODIC_INTERVAL);
        ticker.tick().await; // first tick is immediate
        loop {
            ticker.tick().await;
            if let Err(e) = self.periodic_reprobe_all().await {
                tracing::error!("[CONNECTIVITY] Periodic re-probe error: {}", e);
            }
        }
    }

    async fn periodic_reprobe_all(&self) -> Result<()> {
        let servers = self.server_repository.list().await?;
        for s in servers {
            if !s.is_running() { continue; }
            // Read current connectivity_state to find the public_ip / game_port
            let row: Option<(Value,)> = sqlx::query_as(
                "SELECT connectivity_state FROM servers WHERE id = $1"
            )
            .bind(s.id)
            .fetch_optional(&self.pool)
            .await?;
            let state: ConnectivityState = row
                .and_then(|(v,)| serde_json::from_value(v).ok())
                .unwrap_or_default();
            let public_ip = state.details.get("public_ip")
                .and_then(|v| v.as_str())
                .map(|s| s.to_string());
            let public_ip = match public_ip {
                Some(p) => p,
                None => continue, // no public IP known yet — agent has not reported
            };
            let _ = self.probe_and_classify(s.id, s.node_id, &public_ip, s.port as u16, &s.mc_loader, &state).await;
        }
        Ok(())
    }

    /// WS-dispatched: agent sent a `ConnectivityReport`. Persist + probe + classify + maybe auto-fix.
    pub async fn handle_agent_diagnostics(
        &self,
        server_id: Uuid,
        node_id: Uuid,
        diagnostics: Value,
    ) -> Result<()> {
        // 1) Persist the raw diagnostics into servers.connectivity_state (JSONB)
        let state = ConnectivityState {
            status: "unknown".into(),
            mode: "direct".into(),
            last_probe_at: None,
            failure_mode: None,
            details: diagnostics.clone(),
            last_error: None,
        };
        self.write_connectivity_state(server_id, &state).await?;

        // 2) Append a connectivity_audit_log row
        self.append_audit(server_id, Some(node_id), "connectivity.diagnostics", None, "ok",
            json!({ "diagnostics": diagnostics })).await?;

        // 3) Trigger the probe — public_ip from diagnostics, port from servers table
        let server = self.server_repository.find_by_id(&server_id).await?
            .ok_or_else(|| anyhow!("Server {} not found", server_id))?;
        let public_ip = diagnostics.get("public_ip").and_then(|v| v.as_str())
            .ok_or_else(|| anyhow!("No public_ip in diagnostics"))?;
        self.probe_and_classify(server_id, server.node_id, public_ip,
            server.port as u16, &server.mc_loader, &state).await?;
        Ok(())
    }

    /// WS-dispatched: agent reported back a `ConnectivityFixResult`. Persist + maybe re-probe.
    pub async fn handle_fix_result(
        &self,
        request_id: Uuid,
        server_id: Uuid,
        action: &str,
        success: bool,
        command: &str,
        output: &str,
        details: Option<Value>,
    ) -> Result<()> {
        self.append_audit(
            server_id, None, action, Some(command),
            if success { "ok" } else { "failed" },
            json!({ "request_id": request_id, "output": output, "details": details }),
        ).await?;

        if success {
            // Re-probe after a successful fix
            let server = self.server_repository.find_by_id(&server_id).await?
                .ok_or_else(|| anyhow!("Server {} not found", server_id))?;
            let state = self.read_connectivity_state(server_id).await?;
            let public_ip = state.details.get("public_ip").and_then(|v| v.as_str())
                .map(|s| s.to_string());
            if let Some(public_ip) = public_ip {
                self.probe_and_classify(server_id, server.node_id, &public_ip,
                    server.port as u16, &server.mc_loader, &state).await?;
            }
        }
        Ok(())
    }

    /// REST-dispatched: manual "Reachable" button. Runs the same probe pipeline.
    pub async fn probe_server(&self, server_id: Uuid) -> Result<ConnectivityState> {
        let server = self.server_repository.find_by_id(&server_id).await?
            .ok_or_else(|| anyhow!("Server {} not found", server_id))?;
        let state = self.read_connectivity_state(server_id).await?;
        let public_ip = state.details.get("public_ip").and_then(|v| v.as_str())
            .map(|s| s.to_string());
        let public_ip = public_ip
            .ok_or_else(|| anyhow!("No public_ip recorded yet — agent has not reported diagnostics"))?;
        self.probe_and_classify(server_id, server.node_id, &public_ip,
            server.port as u16, &server.mc_loader, &state).await?;
        self.read_connectivity_state(server_id).await
    }

    async fn probe_and_classify(
        &self,
        server_id: Uuid,
        node_id: Option<Uuid>,
        public_ip: &str,
        port: u16,
        mc_loader: &str,
        state: &ConnectivityState,
    ) -> Result<()> {
        let probe = if mc_loader.eq_ignore_ascii_case("bedrock") {
            probe_bedrock_edition(public_ip, port, PROBE_TIMEOUT).await
        } else {
            probe_java_edition(public_ip, port, PROBE_TIMEOUT).await
        };

        let new_state = match probe {
            Ok(p) if p.tcp_ok && p.handshake_ok && p.status_ok => {
                ConnectivityState {
                    status: "reachable".into(),
                    mode: "direct".into(),
                    last_probe_at: Some(Utc::now()),
                    failure_mode: None,
                    details: merge_details(&state.details, &json!({
                        "version": p.version, "players_online": p.players_online,
                        "players_max": p.players_max, "latency_ms": p.latency_ms,
                    })),
                    last_error: None,
                }
            }
            Ok(p) | Err(_) => {
                let failure = self.classify_failure(&state.details, public_ip, port, node_id).await;
                let was_previous_unreachable = state.status == "unreachable";
                let new = ConnectivityState {
                    status: "unreachable".into(),
                    mode: "direct".into(),
                    last_probe_at: Some(Utc::now()),
                    failure_mode: Some(failure.as_str().to_string()),
                    details: state.details.clone(),
                    last_error: Some(format!("Probe failed: {:?}", p.as_ref().err())),
                };
                // Dispatch auto-fix if applicable AND mode flipped Reachable→Unreachable
                if failure.is_auto_fixable() && !was_previous_unreachable {
                    if let Some(nid) = node_id {
                        if let Err(e) = self.dispatch_fix(nid, server_id, &failure, port).await {
                            tracing::error!("[CONNECTIVITY] Auto-fix dispatch failed: {}", e);
                        }
                    }
                }
                new
            }
        };
        self.write_connectivity_state(server_id, &new_state).await
    }

    async fn classify_failure(
        &self,
        diagnostics: &Value,
        public_ip: &str,
        port: u16,
        _node_id: Option<Uuid>,
    ) -> FailureMode {
        // Order: most-specific first.
        // 1) is_cgn_suspect → CGNAT_DETECTED
        if diagnostics.get("is_cgn_suspect").and_then(|v| v.as_bool()).unwrap_or(false) {
            return FailureMode::CgnatDetected;
        }
        // 2) port_bound == false → PORT_NOT_BOUND
        if diagnostics.get("port_bound").and_then(|v| v.as_bool()) == Some(false) {
            return FailureMode::PortNotBound;
        }
        // 3) firewall_active == true AND no public IP fallback (we already have a public_ip)
        //    → HOST_FIREWALL_BLOCKED
        if diagnostics.get("firewall_active").and_then(|v| v.as_bool()).unwrap_or(false) {
            return FailureMode::HostFirewallBlocked;
        }
        // 4) default → UPNP_UNAVAILABLE (we have a public IP and no other explanation;
        //    probably ISP/NAT issue; the user can enable UPnP or wait for relay)
        let _ = (public_ip, port); // suppress unused warnings
        FailureMode::UpnpUnavailable
    }

    async fn dispatch_fix(
        &self,
        node_id: Uuid,
        server_id: Uuid,
        failure: &FailureMode,
        port: u16,
    ) -> Result<()> {
        let action = match failure {
            FailureMode::PortNotBound => "recreate_container_with_port_bindings",
            FailureMode::HostFirewallBlocked => "firewall.open_port",
            FailureMode::CgnatDetected => return Ok(()), // never auto-fix
            FailureMode::UpnpUnavailable => "upnp.add_mapping",
        };
        let params = match failure {
            FailureMode::HostFirewallBlocked => json!({ "port": port, "proto": "tcp", "server_id": server_id }),
            FailureMode::UpnpUnavailable => json!({ "port": port, "proto": "tcp", "server_id": server_id, "lease_secs": 3600 }),
            _ => json!({ "server_id": server_id }),
        };
        let request_id = Uuid::new_v4();
        self.node_connection_manager.send_to_node(&node_id, &NodeMessage::ConnectivityFixRequest {
            request_id, server_id, action: action.to_string(), params,
        }).await?;
        self.append_audit(server_id, Some(node_id), "connectivity.fix_dispatched", Some(action), "attempted",
            json!({ "request_id": request_id, "action": action }),
        ).await?;
        Ok(())
    }

    // ----- State persistence (direct sqlx on the servers.connectivity_state column) -----
    async fn write_connectivity_state(&self, server_id: Uuid, state: &ConnectivityState) -> Result<()> {
        let v = serde_json::to_value(state)?;

            .bind(v)
            .bind(Utc::now().naive_utc())
            .bind(server_id)
            .execute(&self.pool)
            .await?;
        Ok(())
    }

    async fn read_connectivity_state(&self, server_id: Uuid) -> Result<ConnectivityState> {
        let row: Option<(Value,)> = sqlx::query_as(
            "SELECT connectivity_state FROM servers WHERE id = $1"
        ).bind(server_id).fetch_optional(&self.pool).await?;
        Ok(row.and_then(|(v,)| serde_json::from_value(v).ok()).unwrap_or_default())
    }

    async fn append_audit(

    }

fn merge_details(a: &Value, b: &Value) -> Value {
    let mut merged = a.clone();

        for (k, v) in mb { ma.insert(k.clone(), v.clone()); }
    }
    merged
}

// Minecraft Java Edition SLP probe (D-03 protocol-aware)
// ============================================================
pub async fn probe_java_edition(host: &str, port: u16, timeout: Duration) -> Result<ProbeResult> {
    let mut stream = tokio::time::timeout(timeout, TcpStream::connect((host, port))).await
        .map_err(|_| anyhow!("TCP connect timeout"))?
        .map_err(|e| anyhow!("TCP connect failed: {}", e))?;

    // Handshake packet: [length][packet_id=0x00][protocol=-1][host=string][port=u16][next_state=1]
    let handshake = build_slp_packet(&[
        &encode_varint(0x00),
        &encode_varint(-1),
        &encode_string(host),
        &encode_ushort(port),
        &encode_varint(1),
    ]);
    stream.write_all(&handshake).await.map_err(|e| anyhow!("Handshake write: {}", e))?;

    // Status request: [length][packet_id=0x00]
    let status_req = build_slp_packet(&[&encode_varint(0x00)]);
    stream.write_all(&status_req).await.map_err(|e| anyhow!("Status write: {}", e))?;

    // Read length-prefixed response
    let packet_len = read_varint_async(&mut stream).await.map_err(|e| anyhow!("Read length: {}", e))?;
    let mut buf = vec![0u8; packet_len as usize];
    stream.read_exact(&mut buf).await.map_err(|e| anyhow!("Read body: {}", e))?;
    let packet_id = read_varint_sync(&buf).map_err(|e| anyhow!("Parse packet id: {}", e))?;
    if packet_id != 0x00 { return Ok(ProbeResult { tcp_ok: true, handshake_ok: true, status_ok: false, version: None, players_online: None, players_max: None, latency_ms: None }); }
    let json_str = read_string_sync(&buf).map_err(|e| anyhow!("Parse json: {}", e))?;
    let v: Value = serde_json::from_str(&json_str).map_err(|e| anyhow!("JSON parse: {}", e))?;
    let version = v.get("version").and_then(|v| v.get("name")).and_then(|v| v.as_str()).map(String::from);
    let players_online = v.get("players").and_then(|v| v.get("online")).and_then(|v| v.as_i64()).map(|n| n as i32);
    let players_max = v.get("players").and_then(|v| v.get("max")).and_then(|v| v.as_i64()).map(|n| n as i32);
    Ok(ProbeResult {
        tcp_ok: true, handshake_ok: true, status_ok: true,
        version, players_online, players_max, latency_ms: None,
    })
}

const RAKNET_MAGIC: [u8; 16] = [
    0x00, 0xff, 0xff, 0x00, 0xfe, 0xfe, 0xfe, 0xfe,
    0xfd, 0xfd, 0xfd, 0xfd, 0x12, 0x34, 0x56, 0x78,
];

pub async fn probe_bedrock_edition(host: &str, port: u16, timeout: Duration) -> Result<ProbeResult> {
    let socket = UdpSocket::bind("0.0.0.0:0").await.map_err(|e| anyhow!("UDP bind: {}", e))?;
    socket.connect((host, port)).await.map_err(|e| anyhow!("UDP connect: {}", e))?;
    let client_time = std::time::SystemTime::now()
        .duration_since(std::time::UNIX_EPOCH).unwrap().as_millis() as u64;
    let mut pkt = vec![0x01u8];
    pkt.extend_from_slice(&client_time.to_be_bytes());
    pkt.extend_from_slice(&RAKNET_MAGIC);
    pkt.extend_from_slice(&0x0123456789abcdefu64.to_be_bytes());
    tokio::time::timeout(timeout, socket.send(&pkt)).await
        .map_err(|_| anyhow!("UDP send timeout"))?
        .map_err(|e| anyhow!("UDP send: {}", e))?;
    let mut resp = [0u8; 1500];
    let len = tokio::time::timeout(timeout, socket.recv(&mut resp)).await
        .map_err(|_| anyhow!("UDP recv timeout"))?
        .map_err(|e| anyhow!("UDP recv: {}", e))?;
    let resp = &resp[..len];
    if resp[0] != 0x1c {
        return Ok(ProbeResult { tcp_ok: true, handshake_ok: false, status_ok: false, version: None, players_online: None, players_max: None, latency_ms: None });
    }
    Ok(ProbeResult { tcp_ok: true, handshake_ok: true, status_ok: true, version: None, players_online: None, players_max: None, latency_ms: None })
}

// SLP / Minecraft wire-format helpers
fn encode_varint(v: i32) -> Vec<u8> {
    let mut buf = Vec::with_capacity(5);
    let mut n = v as u32;
    loop {
        if n & !0x7F == 0 { buf.push(n as u8); return buf; }
        buf.push((n as u8 & 0x7F) | 0x80);
        n >>= 7;
    }
}
fn encode_ushort(v: u16) -> Vec<u8> { v.to_be_bytes().to_vec() }
fn encode_string(s: &str) -> Vec<u8> {
    let b = s.as_bytes();
    let mut out = encode_varint(b.len() as i32);
    out.extend_from_slice(b);
    out
}
fn build_slp_packet(parts: &[&[u8]]) -> Vec<u8> {
    let body_len: usize = parts.iter().map(|p| p.len()).sum();
    let mut pkt = encode_varint(body_len as i32);
    for p in parts { pkt.extend_from_slice(p); }
    pkt
}
async fn read_varint_async<R: tokio::io::AsyncReadExt + Unpin>(r: &mut R) -> std::io::Result<i32> {
    let mut num: i32 = 0; let mut shift = 0; loop {
        let mut b = [0u8; 1]; r.read_exact(&mut b).await?;
        num |= ((b[0] & 0x7F) as i32) << shift;
        if b[0] & 0x80 == 0 { return Ok(num); }
        shift += 7; if shift > 35 { return Err(std::io::Error::new(std::io::ErrorKind::InvalidData, "varint too long")); }
    }
}
fn read_varint_sync(buf: &[u8]) -> std::io::Result<i32> {
    let mut num: i32 = 0; let mut shift = 0; let mut i = 0; loop {
        if i >= buf.len() { return Err(std::io::Error::new(std::io::ErrorKind::UnexpectedEof, "varint eof")); }
        let b = buf[i]; i += 1;
        num |= ((b & 0x7F) as i32) << shift;
        if b & 0x80 == 0 { return Ok(num); }
        shift += 7;
    }
}
fn read_string_sync(buf: &[u8]) -> std::io::Result<String> {
    let len = read_varint_sync(buf)? as usize;
    let start = varint_size(buf);
    if buf.len() < start + len { return Err(std::io::Error::new(std::io::ErrorKind::UnexpectedEof, "string eof")); }
    Ok(String::from_utf8_lossy(&buf[start..start+len]).to_string())
}
fn varint_size(buf: &[u8]) -> usize {
    let mut i = 0; while i < buf.len() && buf[i] & 0x80 != 0 { i += 1; } i + 1
}
```
