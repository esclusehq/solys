    pub async fn update_profile(
        State(state): State<ApiState>,
        auth_user: AuthUser,
        Json(req): Json<UpdateProfileRequest>,
    ) -> Result<axum::response::Response, ApiError> {
        let repo = SqlxUserRepository::new(state.pool.clone());

        let mut user = repo.find_by_id(auth_user.user_id)
            .await
            .map_err(|e| ApiError::new("INTERNAL_ERROR", &e.to_string()))?
            .ok_or_else(|| ApiError::new("NOT_FOUND", "User not found"))?;

        if let Some(display_name) = req.display_name {
            // Trim whitespace, disallow empty strings after trim
            let trimmed = display_name.trim().to_string();
            if !trimmed.is_empty() {
                user.display_name = Some(trimmed);
            }
        }
        if let Some(avatar_url) = req.avatar_url {
            let trimmed = avatar_url.trim().to_string();
            if !trimmed.is_empty() {
                user.avatar_url = Some(trimmed);
            }
        }

        let updated = repo.update(&user)
            .await
            .map_err(|e| ApiError::new("INTERNAL_ERROR", &e.to_string()))?;

        let response = serde_json::json!({
            "id": updated.id,
            "email": updated.email,
            "display_name": updated.display_name,
            "avatar_url": updated.avatar_url,
        });

        Ok(ApiResponse::new(response).into_response())
    }

    ```rust
    pub async fn request_account_deletion(
        State(state): State<ApiState>,
        auth_user: AuthUser,
        Json(req): Json<DeleteAccountRequest>,
    ) -> Result<axum::response::Response, ApiError> {
        // D-09: Verify "DELETE" text confirmation
        if req.confirm_text != "DELETE" {
            return Err(ApiError::new("VALIDATION_ERROR", "Type DELETE to confirm"));
        }

        let repo = SqlxUserRepository::new(state.pool.clone());

        let user = repo.find_by_id(auth_user.user_id)
            .await
            .map_err(|e| ApiError::new("INTERNAL_ERROR", &e.to_string()))?
            .ok_or_else(|| ApiError::new("NOT_FOUND", "User not found"))?;

        // D-09: Re-authenticate via password verification
        if !crate::domain::auth::service::PasswordService::verify(&req.password, &user.password_hash) {
            return Err(ApiError::new("UNAUTHORIZED", "Invalid password"));
        }

        // D-07: Schedule soft-deletion — request-date semantics
        // scheduled_deletion_at stores the request timestamp; grace period end = scheduled_at + 14 days
        let scheduled_at = chrono::Utc::now();
        repo.schedule_deletion(auth_user.user_id, scheduled_at)
            .await
            .map_err(|e| ApiError::new("INTERNAL_ERROR", &e.to_string()))?;

        tracing::info!("Account deletion scheduled for user {} at {}", auth_user.user_id, scheduled_at);

        Ok(ApiResponse::success(serde_json::json!({
            "message": "Account deletion scheduled. You have 14 days to cancel.",
            "scheduled_deletion_at": scheduled_at,
        })).into_response())
    }
    ```

    **E) `transfer_ownership` handler** (D-08 — actual server ownership transfer):
    ```rust
    pub async fn transfer_ownership(
        State(state): State<ApiState>,
        auth_user: AuthUser,
        Json(req): Json<TransferOwnershipRequest>,
    ) -> Result<axum::response::Response, ApiError> {
        // D-08: Find target user by email
        let user_repo = SqlxUserRepository::new(state.pool.clone());

        let target = user_repo.find_by_email(&req.target_email)
            .await
            .map_err(|e| ApiError::new("INTERNAL_ERROR", &e.to_string()))?
            .ok_or_else(|| ApiError::new("NOT_FOUND", "Target user not found"))?;

        if target.id == auth_user.user_id {
            return Err(ApiError::new("VALIDATION_ERROR", "Cannot transfer ownership to yourself"));
        }

        // Transfer servers: query all servers owned by current user, reassign to target
        let server_repo = SqlxServerRepository::new(state.pool.clone());
        let owned_servers = server_repo.find_by_user_id(auth_user.user_id)
            .await
            .map_err(|e| ApiError::new("INTERNAL_ERROR", &e.to_string()))?;

        if owned_servers.is_empty() {
            return Err(ApiError::new("NOT_FOUND", "No servers found to transfer"));
        }

        let transfer_count = owned_servers.len();
        for server in &owned_servers {
            let mut updated_server = server.clone();
            updated_server.user_id = Some(target.id);
            server_repo.update(&updated_server)
                .await
                .map_err(|e| ApiError::new("INTERNAL_ERROR", &e.to_string()))?;
        }

        tracing::info!(
            "Ownership transferred: {} servers from user {} -> user {}",
            transfer_count, auth_user.user_id, target.id
        );

        Ok(ApiResponse::success(serde_json::json!({
            "message": format!("{} servers transferred to {}", transfer_count, target.email),
            "servers_transferred": transfer_count,
            "target_user_id": target.id,
            "target_email": target.email,
        })).into_response())
    }
    ```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Client → /auth/* API | Unauthenticated (register, login, forgot-password) and authenticated (me, profile, account) endpoints |
| AuthUser middleware → Handlers | Authenticated user identity from JWT/cookie |
| /auth/account/delete handler | Accepts password + confirmation text |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-53-03-01 | Spoofing | PUT /auth/profile | mitigate | AuthUser middleware ensures only authenticated user can update own profile |
| T-53-03-02 | Tampering | POST /auth/account/delete | mitigate | Password re-verification (D-09) + "DELETE" text confirmation prevent accidental/malicious deletion |
| T-53-03-03 | Information Disclosure | GET /auth/login-history | mitigate | Query scoped to `WHERE user_id = $1` — only returns own login history |
| T-53-03-04 | Repudiation | login_history INSERT | mitigate | Backend-driven insertion captures IP + user-agent at auth time; immutable audit-style table (no update/delete by user) |
| T-53-03-05 | Denial of Service | Login history table growth | mitigate | D-06: 90-day retention with cron cleanup; query limited to 100 rows |
| T-53-03-06 | Information Disclosure | X-Forwarded-For header | accept | Header comes from reverse proxy (Caddy) — trusted network; if spoofed, IP stored is inaccurate but not security-critical for login history |
</threat_model>
