  <action>
    Create a new background service file that performs two periodic cleanup tasks:

    1. **Account hard deletion** — finds users where `scheduled_deletion_at` is set AND the 14-day grace period has expired, then permanently deletes them (D-07).
    2. **Login history cleanup** — deletes login_history records older than 90 days (D-06).

    ```rust
    use std::sync::Arc;
    use std::time::Duration;
    use sqlx::PgPool;
    use chrono::Utc;
    use tracing;

    use crate::domain::user::sqlx_repository::SqlxUserRepository;
    use crate::domain::user::repository::UserRepository;

    pub struct DeletionCleanupService {
        pool: PgPool,
        interval: Duration,
    }

    impl DeletionCleanupService {
        pub fn new(pool: PgPool) -> Self {
            Self {
                pool,
                interval: Duration::from_secs(3600), // Run every hour
            }
        }

        pub async fn run(&self) {
            let mut interval = tokio::time::interval(self.interval);
            loop {
                interval.tick().await;
                tracing::info!("Running deletion cleanup cycle...");

                if let Err(e) = self.cleanup_expired_deletions().await {
                    tracing::error!("Failed to cleanup expired deletions: {}", e);
                }

                if let Err(e) = self.cleanup_old_login_history().await {
                    tracing::error!("Failed to cleanup old login history: {}", e);
                }
            }
        }

        async fn cleanup_expired_deletions(&self) -> Result<(), Box<dyn std::error::Error + Send + Sync>> {
            let repo = SqlxUserRepository::new(self.pool.clone());
            let users = repo.find_by_deletion_scheduled().await?;

            let now = Utc::now();
            let mut deleted_count = 0;

            for user in users {
                if let Some(scheduled_at) = user.scheduled_deletion_at {
                    // Check if 14-day grace period has expired (D-07)
                    let grace_end = scheduled_at + chrono::Duration::days(14);
                    if now > grace_end {
                        tracing::info!(
                            "Permanently deleting user {} (scheduled: {}, grace ended: {})",
                            user.id, scheduled_at, grace_end
                        );
                        repo.hard_delete(user.id).await?;
                        deleted_count += 1;
                    }
                }
            }

            if deleted_count > 0 {
                tracing::info!("Deleted {} expired accounts", deleted_count);
            }

            Ok(())
        }

        async fn cleanup_old_login_history(&self) -> Result<(), Box<dyn std::error::Error + Send + Sync>> {
            // D-06: Delete login history older than 90 days
            let cutoff = Utc::now() - chrono::Duration::days(90);
            let result = sqlx::query("DELETE FROM login_history WHERE created_at < $1")
                .bind(cutoff)
                .execute(&self.pool)
                .await?;

            if result.rows_affected() > 0 {
                tracing::info!("Cleaned up {} old login history records", result.rows_affected());
            }

            Ok(())
        }
    }
    ```
