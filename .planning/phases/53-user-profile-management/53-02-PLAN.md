    **B) Implement the four new methods from the trait:**

    ```rust
    async fn find_by_deletion_scheduled(&self) -> Result<Vec<User>, Box<dyn std::error::Error + Send + Sync>> {
        let results = sqlx::query_as::<_, User>(
            "SELECT * FROM users WHERE scheduled_deletion_at IS NOT NULL AND deleted_at IS NULL ORDER BY scheduled_deletion_at ASC"
        )
        .fetch_all(&self.pool)
        .await?;
        Ok(results)
    }

    async fn schedule_deletion(&self, id: Uuid, scheduled_at: DateTime<Utc>) -> Result<(), Box<dyn std::error::Error + Send + Sync>> {
        sqlx::query("UPDATE users SET scheduled_deletion_at = $2, updated_at = NOW() WHERE id = $1 AND deleted_at IS NULL")
            .bind(id)
            .bind(scheduled_at)
            .execute(&self.pool)
            .await?;
        Ok(())
    }

    async fn cancel_deletion(&self, id: Uuid) -> Result<(), Box<dyn std::error::Error + Send + Sync>> {
        sqlx::query("UPDATE users SET scheduled_deletion_at = NULL, updated_at = NOW() WHERE id = $1 AND deleted_at IS NULL")
            .bind(id)
            .execute(&self.pool)
            .await?;
        Ok(())
    }

    async fn hard_delete(&self, id: Uuid) -> Result<(), Box<dyn std::error::Error + Send + Sync>> {
        sqlx::query("DELETE FROM users WHERE id = $1")
            .bind(id)
            .execute(&self.pool)
            .await?;
        Ok(())
    }
    ```

    Add `use chrono::{DateTime, Utc};` at the top of the file if not already imported.
