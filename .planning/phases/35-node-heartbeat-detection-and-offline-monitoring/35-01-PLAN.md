```rust
/// Returns true if node has recent heartbeat (within OFFLINE_THRESHOLD seconds)
pub fn is_online(&self) -> bool {
    const OFFLINE_THRESHOLD_SECS: i64 = 60; // configurable
    if let Some(last_seen) = self.last_seen {
        let age = Utc::now().signed_duration_since(last_seen).num_seconds();
        return age < OFFLINE_THRESHOLD_SECS && self.status == "online";
    }
    false
}

/// Returns true if node has no recent heartbeat
pub fn is_offline(&self) -> bool {
    !self.is_online()
}
```

```rust
use tokio::time::sleep;
use std::time::Duration;

fn calculate_backoff(attempt: u32) -> Duration {
    // 1s -> 2s -> 4s -> 8s -> 16s -> 32s -> 64s -> max 300s
    let base = 2_f64.powi(attempt as i32);
    let seconds = base.min(300.0);
    Duration::from_secs(seconds as u64)
}

async fn execute_with_backoff<F, T, E>(
    mut attempt: u32,
    max_attempts: u32,
    operation: F,
) -> Result<T, E>
where
    F: Fn() -> Fut<T, E>,
    Fut: std::future::Future<Output = Result<T, E>>,
{
    loop {
        match operation().await {
            Ok(result) => return Ok(result),
            Err(e) if attempt >= max_attempts => return Err(e),
            Err(e) => {
                let backoff = calculate_backoff(attempt);
                tracing::warn!("Operation failed (attempt {}), retrying in {}s: {}", 
                    attempt + 1, backoff.as_secs(), e);
                sleep(backoff).await;
                attempt += 1;
            }
        }
    }
}
```

Use this in critical operations like get_status, get_metrics, start_server with max 5 attempts.

```rust
use std::cell::RefCell;
use std::collections::HashMap;

thread_local! {
    static LAST_LOG_TIME: RefCell<HashMap<String, std::time::Instant>> = RefCell::new(HashMap::new());
}

fn should_log(key: &str, interval_secs: u64) -> bool {
    let now = std::time::Instant::now();
    LAST_LOG_TIME.with(|times| {
        let mut times = times.borrow_mut();
        if let Some(last) = times.get(key) {
            if now.duration_since(*last).as_secs() < interval_secs {
                return false;
            }
        }
        times.insert(key.to_string(), now);
        true
    })
}
```
