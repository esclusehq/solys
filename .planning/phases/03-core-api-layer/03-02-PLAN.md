From code review (lines 225-242):
```rust
async fn get_server(
    State(state): State<ApiState>,
    auth_user: AuthUser,
    Path(id): Path<Uuid>,
) -> Result<impl IntoResponse, String> {
    let repo = SqlxServerRepository::new(state.pool.clone());
    let server = repo.find_by_id(id)
        .await
        .map_err(|e| e.to_string())?
        .ok_or_else(|| "Server not found".to_string())?;
    
    // Check tenant access
    if server.user_id != auth_user.tenant_id {
        return Err("Access denied".to_string());
    }
    
    Ok(Json(ApiResponse::success(server)))
}
```

Ownership check is properly enforced. No changes needed.
