Create release/package/install.ps1 PowerShell installer section for service:

```powershell
$nssmPath = "$env:ProgramFiles\solys\nssm.exe"
# ... download and install nssm ...
nssm install solys "$env:ProgramFiles\solys\solys.exe"
nssm set solys AppDirectory "$env:ProgramFiles\solys"
nssm set solys Start SERVICE_AUTO_START
nssm set solys RestartServiceDelay 10000
nssm set solys AppRestartExitAction RESTART
Start-Service solys
```
