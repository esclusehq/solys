       ```javascript
       import { useState, useEffect, useCallback } from 'react';
       import { relayApi } from '../lib/api';

       export function useConnectivity(serverId, { pollIntervalMs = 15000 } = {}) {
         const [state, setState] = useState({
           status: 'disabled',         // disabled | connecting | connected | stale | failed
           mode: null,                 // null | "relay" | "direct" | "auto"
           lastConnected: null,        // ISO timestamp
           lastDisconnected: null,     // ISO timestamp
           reason: null,               // disconnect reason string
           loading: true,
           error: null,
         });

         const refresh = useCallback(async () => {
           try {
             const [modeRes, healthRes] = await Promise.all([
               relayApi.getMode(serverId),
               relayApi.getTunnelHealth(serverId),
             ]);
             setState({
               status: healthRes.data.relay_status,
               mode: modeRes.data.mode,
               lastConnected: healthRes.data.last_connected_at,
               lastDisconnected: healthRes.data.last_disconnected_at,
               reason: healthRes.data.disconnect_reason,
               loading: false,
               error: null,
             });
           } catch (e) {
             setState((s) => ({ ...s, loading: false, error: e.message }));
           }
         }, [serverId]);

         useEffect(() => {
           refresh();
           const id = setInterval(refresh, pollIntervalMs);
           return () => clearInterval(id);
         }, [refresh, pollIntervalMs]);

         const setMode = useCallback(async (newMode) => {
           await relayApi.setMode(serverId, newMode);
           await refresh();
         }, [serverId, refresh]);

         return { ...state, setMode, refresh };
       }
       ```
