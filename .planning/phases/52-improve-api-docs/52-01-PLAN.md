    ```typescript
    import { defineLoader } from 'vitepress'

    export interface OpenApiSchema {
      type: string
      properties: Record<string, {
        type?: string
        description?: string
        $ref?: string
        enum?: string[]
        default?: unknown
      }>
      required?: string[]
    }

    declare const data: Record<string, OpenApiSchema>
    export { data }

    export default defineLoader({
      async load(): Promise<Record<string, OpenApiSchema>> {
        try {
          const res = await fetch('https://api.esluce.com/openapi.json', {
            signal: AbortSignal.timeout(10000)
          })
          if (!res.ok) {
            console.warn(`OpenAPI fetch failed with status ${res.status}; using empty schemas`)
            return {}
          }
          const spec: { components?: { schemas?: Record<string, OpenApiSchema> } } = await res.json()
          return spec.components?.schemas ?? {}
        } catch (err) {
          console.warn(`Failed to fetch OpenAPI spec: ${err}; using empty schemas`)
          return {}
        }
      },
    })
    ```

    Key details:
    - Use `fetch` (Node.js 18+ built-in, VitePress >= 1.6 requires Node.js 18)
    - Set 10s timeout via `AbortSignal.timeout`
    - Graceful fallback — return `{}` on failure, never crash the build
    - Interface `OpenApiSchema` is exported so Vue components can import it for type safety
    - Per D-02, this fetches at BUILD TIME only, not during dev server hot-reload

    Do NOT install any extra npm packages — VitePress's built-in `defineLoader` is already available.

    Create the directory `docs/.vitepress/components/` and create TWO Vue components:

    **Component 1: `OpenApiSchema.vue`**
    Renders a schema table by looking up a `$ref` path (e.g. `#/components/schemas/CreateServerRequest`) in the data loader's schema map.

    ```vue
    <script setup lang="ts">
    import { data as schemas } from '../loaders/openapi.data'

    const props = defineProps<{
      ref: string          // e.g. "#/components/schemas/CreateServerRequest"
      type?: 'request' | 'response'
    }>()

    const schemaName = props.ref.split('/').pop()!
    const schema = schemas[schemaName]
    const properties = schema?.properties || {}
    const required = schema?.required || []
    </script>

    <template>
      <div class="schema-table-wrapper">
        <div v-if="props.type === 'request'" class="schema-context-label">Request Body Schema</div>
        <div v-else-if="props.type === 'response'" class="schema-context-label">Response Schema</div>
        <table v-if="schema" class="schema-table">
          <thead>
            <tr>
              <th>Field</th>
              <th>Type</th>
              <th>Required</th>
              <th>Description</th>
            </tr>
          </thead>
          <tbody>
            <tr v-for="(prop, name) in properties" :key="String(name)">
              <td><code>{{ name }}</code></td>
              <td>{{ prop.type || prop.$ref?.split('/').pop() || 'object' }}</td>
              <td>{{ required.includes(String(name)) ? 'Yes' : 'No' }}</td>
              <td>
                {{ prop.description || '-' }}
                <span v-if="prop.enum" class="schema-enum-hint">Possible values: {{ prop.enum.join(', ') }}</span>
              </td>
            </tr>
          </tbody>
        </table>
        <p v-else class="schema-missing">Schema "{{ schemaName }}" not found in OpenAPI spec. See <StaticSchema documentation for manual schema definitions.</p>
      </div>
    </template>
    ```

    **Component 2: `StaticSchema.vue`**
    Renders a schema table from an inline JSON object passed as a prop. For endpoints NOT in the OpenAPI spec.

      type?: 'request' | 'response'
    }>()

    <template>
      <div class="schema-table-wrapper">
        <div v-if="props.type === 'request'" class="schema-context-label">Request Body Schema</div>
        <div v-else-if="props.type === 'response'" class="schema-context-label">Response Schema</div>
        <table v-if="props.schema" class="schema-table">
          <thead>
            <tr>
              <th>Field</th>
              <th>Type</th>
              <th>Required</th>
              <th>Description</th>
            </tr>
          </thead>
          <tbody>
            <tr v-for="(prop, name) in properties" :key="String(name)">
              <td><code>{{ name }}</code></td>
              <td>{{ prop.type || prop.$ref?.split('/').pop() || 'object' }}</td>
              <td>{{ required.includes(String(name)) ? 'Yes' : 'No' }}</td>

              </td>
            </tr>
          </tbody>
        </table>
        <p v-else class="schema-missing">No schema provided.</p>
      </div>
    </template>
