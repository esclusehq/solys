    2. Support `search_mods(query, game_version, mod_loader, offset, sort)` returning search results
    3. Support `get_mod_versions(mod_id)` returning available versions with file download URLs

    ```rust
    use serde::Deserialize;
    use anyhow::{Result, Context};

    const CURSEFORGE_API_BASE: &str = "https://api.curseforge.com/v1";

    /// Client for the CurseForge API to search and download Minecraft mods.
    /// Requires an API key set in platform settings (x-api-key header).
    #[derive(Clone)]
    pub struct CurseForgeClient {
        client: reqwest::Client,
        api_key: Option<String>,
    }

    // ── CurseForge API Response Types ────────────────────────────────

    #[derive(Debug, Deserialize)]
    pub struct CurseForgeSearchResponse {
        pub data: Vec<CurseForgeMod>,
        pub pagination: CurseForgePagination,
    }

    #[derive(Debug, Clone, Deserialize)]
    pub struct CurseForgeMod {
        pub id: u64,
        pub slug: String,
        pub name: String,
        pub summary: String,
        pub logo: Option<CurseForgeLogo>,
        pub download_count: u64,
        pub categories: Vec<CurseForgeCategory>,
        pub game_versions: Vec<String>,
        pub latest_files: Vec<CurseForgeFile>,
        pub mod_loader_types: Vec<String>,
        pub game_popularity_rank: Option<u64>,
    }

    #[derive(Debug, Clone, Deserialize)]
    pub struct CurseForgeLogo {
        pub url: String,
        pub thumbnail_url: Option<String>,
    }

    #[derive(Debug, Clone, Deserialize)]
    pub struct CurseForgeCategory {
        pub id: u64,
        pub name: String,
        pub slug: String,
    }

    #[derive(Debug, Clone, Deserialize)]
    pub struct CurseForgePagination {
        pub index: u32,
        pub page_size: u32,
        pub total_count: u32,
    }

    #[derive(Debug, Deserialize)]
    pub struct CurseForgeFileResponse {
        pub data: CurseForgeFile,
    }

    #[derive(Debug, Clone, Deserialize)]
    pub struct CurseForgeFile {
        pub id: u64,
        pub file_name: String,
        pub download_url: String,
        pub file_length: u64,
        pub game_versions: Vec<String>,
        pub release_type: String,
    }

    impl CurseForgeClient {
        pub fn new(api_key: Option<String>) -> Self {
            let client = reqwest::Client::builder()
                .user_agent("DevNode-V2/2.0.0 (github.com/rhnbztnl/DevNode-V2)")
                .build()
                .expect("Failed to build reqwest client");
            Self { client, api_key }
        }

        pub fn set_api_key(&mut self, api_key: Option<String>) {
            self.api_key = api_key;
        }

        fn build_request(&self, path: &str) -> reqwest::RequestBuilder {
            let req = self.client.get(format!("{}{}", CURSEFORGE_API_BASE, path));
            if let Some(key) = &self.api_key {
                req.header("x-api-key", key)
            } else {
                req
            }
        }

        /// Search for mods on CurseForge.
        /// game_id = 432 (Minecraft), class_id optional (6 = mods, 4471 = modpacks)
        pub async fn search_mods(
            &self,
            query: &str,
            game_version: Option<&str>,
            mod_loader: Option<&str>,
            class_id: Option<u32>,
            offset: Option<u32>,
            sort: Option<&str>,
        ) -> Result<CurseForgeSearchResponse> {
            let mut path = format!(
                "/mods/search?gameId=432&searchFilter={}&index={}&pageSize=25",
                urlencoding(query),
                offset.unwrap_or(0),
            );
            if let Some(v) = game_version {
                path.push_str(&format!("&gameVersion={}", urlencoding(v)));
            }
            if let Some(loader) = mod_loader {
                path.push_str(&format!("&modLoaderType={}", loader));
            }
            if let Some(cid) = class_id {
                path.push_str(&format!("&classId={}", cid));
            }
            // sort: "popularity", "downloads", "last_updated", "name", "author", "date_created"
            if let Some(s) = sort {
                path.push_str(&format!("&sortField={}", urlencoding(s)));
            }

            let response = self.build_request(&path)
                .send()
                .await
                .context("Failed to send CurseForge search request")?;

            if !response.status().is_success() {
                anyhow::bail!("CurseForge API returned {}: {}", response.status(), response.text().await?);
            }

            let data: CurseForgeSearchResponse = response
                .json()
                .await
                .context("Failed to parse CurseForge search response")?;

            Ok(data)
        }

        /// Get available versions for a mod
        pub async fn get_mod_files(&self, mod_id: u64) -> Result<Vec<CurseForgeFile>> {
            let response = self.build_request(&format!("/mods/{}/files", mod_id))
                .send()
                .await
                .context("Failed to fetch CurseForge mod files")?;

            if !response.status().is_success() {
                anyhow::bail!("CurseForge API returned {}", response.status());
            }

            #[derive(Deserialize)]
            struct FileListResponse {
                data: Vec<CurseForgeFile>,
            }

            let data: FileListResponse = response.json().await?;
            Ok(data.data)
        }

        /// Get download URL for a specific mod file
        pub async fn get_file_download_url(&self, mod_id: u64, file_id: u64) -> Result<String> {
            let response = self.build_request(&format!("/mods/{}/files/{}", mod_id, file_id))
                .send()
                .await
                .context("Failed to fetch CurseForge file details")?;

            if !response.status().is_success() {
                anyhow::bail!("CurseForge API returned {}", response.status());
            }

            let data: CurseForgeFileResponse = response.json().await?;
            Ok(data.data.download_url)
        }
    }

    /// Percent-encode a string for URL query parameters.
    fn urlencoding(s: &str) -> String {
        use std::fmt::Write;
        let mut result = String::new();
        for byte in s.bytes() {
            match byte {
                b'A'..=b'Z' | b'a'..=b'z' | b'0'..=b'9' | b'-' | b'_' | b'.' | b'~' => {
                    result.push(byte as char);
                }
                b' ' => result.push_str("%20"),
                _ => {
                    write!(&mut result, "%{:02X}", byte).unwrap();
                }
            }
        }
        result
    }
    ```
