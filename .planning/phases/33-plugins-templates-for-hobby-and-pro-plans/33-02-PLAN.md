```javascript
  const response = await api.post(`/servers/${serverId}/install-template`, { template_id: templateId });
```

Create React hook for plugin templates:
- fetchPluginTemplates(gameType?) - fetches available templates
- installTemplate(serverId, templateId) - triggers auto-install
- Plan check: checks user plan from auth store
- Only returns templates for Hobby/Pro plans, null for Starter

```javascript
export function usePluginTemplates() {
  const { user } = useAuthStore();
  const isHobbyPlus = user?.plan === 'hobby' || user?.plan === 'pro' || user?.plan === 'enterprise';
  
  const fetch = async (gameType) => {
    if (!isHobbyPlus) return []; // Starter can't use templates
    return fetchPluginTemplates(gameType);
  };
  
  return { fetchPluginTemplates: fetch, installTemplate };
}
```
